34systemData::initializeData()
35{
36 spdlog::get(m_logName)->info("Running initializeData()");
37
38 // load and check data from GSD
39 parseGSD();
40
41 // initialize D.o.F. parameters
42 m_num_DoF = 6 * m_num_bodies; // D.o.F. are linear and angular positions of body centers
43 m_num_constraints = m_num_bodies; // 1 unit quaternion constraint per body
44

176void
177systemData::velocitiesArticulation()
178{
179 double dimensional_time{m_tau * m_t};
180
181 // ANCHOR: Orientation vectors, q = R_1 - R_3
182 Eigen::Vector3d q = m_positions_particles.segment<3>(3 * 0) - m_positions_particles.segment<3>(3 * 2);
183 q.normalize();
184 Eigen::Vector3d q_tilde = q;
185 q_tilde(2) *= -1;
186
187 // articulation velocity magnitudes
188 const double v1_mag = m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time);
189 const double v3_mag = m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time + m_sys_spec_phase_shift);
190
191 // Zero and then calculate m_velocities_particles_articulation
192 m_velocities_particles_articulation.setZero();
193 m_velocities_particles_articulation.segment<3>(3 * 0).noalias() = v1_mag * q;
194 m_velocities_particles_articulation.segment<3>(3 * 2).noalias() = v3_mag * q;
195 m_velocities_particles_articulation.segment<3>(3 * 3).noalias() = v1_mag * q_tilde;
196 m_velocities_particles_articulation.segment<3>(3 * 5).noalias() = v3_mag * q_tilde;
197}

258void
259systemData::rigidBodyMotionTensors(Eigen::ThreadPoolDevice& device)
260{
261 /* ANCHOR: Compute m_rbm_conn */
262 m_rbm_conn.setZero();
263
264 for (int i = 0; i < m_num_particles; i++)
265 {
266 const int i3{3 * i};
267 const Eigen::Vector3d R_i = m_positions_particles.segment<3>(i3);
268
269 // Calculate which body j particle i belongs to
270 const int body_j = (m_particle_type_id.segment(0, i + 1).array() == 1).count() - 1;
271 const int j6{6 * body_j};
272 const int j7{7 * body_j};
273
274 // (negative) moment arm of particle about its body's locater position
275 Eigen::Vector3d n_dr = m_positions_bodies.segment<3>(j7);
276 n_dr.noalias() -= R_i;
277
278 // skew-symmetric matrix representation of cross product
279 Eigen::Matrix3d n_dr_cross;
280 crossProdMat(n_dr, n_dr_cross);
281
282 // rigid body motion connectivity tensor elements
283 m_rbm_conn.block<3, 3>(j6, i3).noalias() = m_I3; // translation-translation couple
284 m_rbm_conn.block<3, 3>(j6 + 3, i3).noalias() = n_dr_cross; // translation-rotation couple
285
286 if (i == 0)
287 {
288 assert(body_j == 0 && "First particle must be part of body 0.");
289 }
290 if (i == m_num_particles - 1)
291 {
292 assert(body_j + 1 == m_num_bodies && "Last particle must be part of body M.");
293 }
294 }
295
296 /* ANCHOR: Compute m_psi_conv_quat_ang */
297 m_psi_conv_quat_ang.setZero();
298
299 for (int k = 0; k < m_num_bodies; k++)
300 {
301 const int k6{6 * k};
302 const int k7{7 * k};
303
304 // matrix E from quaterion of body k
305 Eigen::Matrix<double, 3, 4> E_theta_k;
306 eMatrix(m_positions_bodies.segment<4>(k7 + 3), E_theta_k);
307
308 // matrix elements of Psi
309 m_psi_conv_quat_ang.block<3, 3>(k6, k7).noalias() = m_I3; // no conversion from linear components
310 m_psi_conv_quat_ang.block<3, 4>(k6 + 3, k7 + 3).noalias() = 2 * E_theta_k; // angular-quaternion velocity couple
311 }
312
313 /* ANCHOR: Compute m_rbm_conn_T_quat */
314 m_rbm_conn_T_quat.noalias() = m_rbm_conn.transpose() * m_psi_conv_quat_ang;
315 m_tens_rbm_conn_T_quat = TensorCast(m_rbm_conn_T_quat);
316}

318void
319systemData::gradientChangeOfVariableTensors(Eigen::ThreadPoolDevice& device)
320{
321 /* ANCHOR: Compute m_conv_body_2_part_dof and m_tens_conv_body_2_part_dof */
322 m_conv_body_2_part_dof.setZero();
323
324 int body_num{-1}; // -1 as first particle should be locater particle and increment this
325 Eigen::Matrix<double, 3, 4> twoE_body;
326 Eigen::Vector4d R_c_body;
327
328 assert(m_particle_type_id(0) == 1 && "First particle index must be locater by convention");
329 for (int j = 0; j < m_num_particles; j++)
330 {
331
332 if (m_particle_type_id(j) == 1)
333 {
334 // Zero out the relevent locater values
335 twoE_body.setZero();
336 R_c_body.setZero();
337 // Increment body count
338 body_num += 1;
339
340 // Get locater position of body i
341 R_c_body.segment<3>(1).noalias() = m_positions_locater_particles.segment<3>(3 * body_num);
342 // Get E matrix representation of body quaternion from m_psi_conv_quat_ang
343 twoE_body.noalias() = m_psi_conv_quat_ang.block<3, 4>(6 * body_num + 3, 7 * body_num + 3);
344
345 // Continue to next loop as all elements are zero
346 continue;
347 }
348
349 // j -> particle number
350 const int j3{3 * j};
351
352 // 4-vector version of particle j position (prepend zero element)
353 Eigen::Vector4d R_j = Eigen::Vector4d::Zero(4, 1);
354 R_j.segment<3>(1).noalias() = m_positions_particles.segment<3>(j3);
355
356 // matrix P_tilde from moment arm
357 const Eigen::Vector4d dr = R_j - R_c_body;
358 Eigen::Matrix<double, 3, 4> P_j_tilde_hold;
359 eMatrix(dr, P_j_tilde_hold);
360 Eigen::Matrix<double, 4, 3> P_j_tilde = P_j_tilde_hold.transpose();
361
362 // change of variables gradient tensor elements
363 m_conv_body_2_part_dof.block<3, 3>(7 * body_num, j3).noalias() = -m_I3; // translation-translation couple
364 m_conv_body_2_part_dof.block<3, 3>(7 * body_num + 4, j3).noalias() =
365 twoE_body * P_j_tilde; // quaternion-rotation couple (first row is zero)
366 }
367 assert(body_num + 1 == m_num_bodies && "Not all bodies were indexed correctly");
368
369 m_tens_conv_body_2_part_dof = TensorCast(m_conv_body_2_part_dof);
370
371 /* ANCHOR : Compute m_rbm_conn_T_quat_grad */
372 m_rbm_conn_T_quat_grad.setZero();
373 body_num = -1;
374
375 assert(m_particle_type_id(0) == 1 && "First particle index must be locater by convention");
376 for (int i = 0; i < m_num_particles; i++)
377 {
378 if (m_particle_type_id(i) == 1)
379 {
380 // Increment body count
381 body_num += 1;
382
383 // Continue to next loop as all elements are zero
384 continue;
385 }
386
387 // get moment arm to locater point from C
388 const Eigen::Matrix3d two_r_cross_mat = -2 * m_rbm_conn.block<3, 3>(6 * body_num + 3, 3 * i);
389
390 // get E matrix representation of body quaternion from m_psi_conv_quat_ang
391 const Eigen::Matrix<double, 3, 4> two_E_body =
392 m_psi_conv_quat_ang.block<3, 4>(6 * body_num + 3, 7 * body_num + 3);
393
394 // get change of variable matrix element D_{\alpha}
395 const Eigen::Matrix<double, 7, 3> n_D_alpha = -m_conv_body_2_part_dof.block<7, 3>(7 * body_num, 3 * i);
396
397 /* ANCHOR: tensor contractions to produce result */
398
399 // convert Eigen::Matrix --> Eigen::Tensor
400 const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_two_r_cross_mat = TensorCast(two_r_cross_mat);
401 const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4>> tens_two_E_body = TensorCast(two_E_body);
402 const Eigen::TensorFixedSize<double, Eigen::Sizes<7, 3>> tens_n_D_alpha = TensorCast(n_D_alpha);
403
404 // 1) Compute result_{i m k} = levi_cevita{l i m} n_D_alpha_{k l}
405 Eigen::array<Eigen::IndexPair<int>, 1> contract_lim_kl = {Eigen::IndexPair<int>(0, 1)}; // {i m k}
406 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 7>> d_rCrossMat_d_xi;
407 d_rCrossMat_d_xi.device(device) = m_levi_cevita.contract(tens_n_D_alpha, contract_lim_kl);
408
409 // 2) Compute result_{i j k} = d_rCrossMat_d_xi_{i m k} two_E_body{m j}
410 Eigen::array<Eigen::IndexPair<int>, 1> contract_imk_mj = {Eigen::IndexPair<int>(1, 0)}; // {i k j}, must shuffle
411 Eigen::array<int, 3> swap_last_two_indices({0, 2, 1});
412
413 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 7, 4>> d_rCrossMat_d_xi_times_two_E_body_preshuffle;
414 d_rCrossMat_d_xi_times_two_E_body_preshuffle.device(device) =
415 d_rCrossMat_d_xi.contract(tens_two_E_body, contract_imk_mj);
416
417 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> d_rCrossMat_d_xi_times_two_E_body;
418 d_rCrossMat_d_xi_times_two_E_body.device(device) =
419 d_rCrossMat_d_xi_times_two_E_body_preshuffle.shuffle(swap_last_two_indices);
420
421 // 3) Compute result_{i j k} = two_r_cross_mat_{i m} Kappa_tilde_{m j k} (same indices as (2))
422 Eigen::array<Eigen::IndexPair<int>, 1> contract_im_mjk = {Eigen::IndexPair<int>(1, 0)}; // {i j k}
423 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> rCrossMat_times_d_E_body_d_xi;
424 rCrossMat_times_d_E_body_d_xi.device(device) = tens_two_r_cross_mat.contract(m_kappa_tilde, contract_im_mjk);
425
426 // Compute and output matrix element
427 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> angular_gradient;
428 angular_gradient.device(device) = -d_rCrossMat_d_xi_times_two_E_body - rCrossMat_times_d_E_body_d_xi;
429
430 // output matrix indices
431 Eigen::array<Eigen::Index, 3> offsets = {3 * i, 7 * body_num + 3, 7 * body_num};
432 Eigen::array<Eigen::Index, 3> extents = {3, 4, 7};
433 m_rbm_conn_T_quat_grad.slice(offsets, extents) = angular_gradient;
434 }
435 assert(body_num + 1 == m_num_bodies && "Not all bodies were indexed correctly");
436}
