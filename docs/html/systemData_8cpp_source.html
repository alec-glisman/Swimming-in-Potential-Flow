261 void
262 systemData::rigidBodyMotionTensors()
263 {
264  /* ANCHOR: Compute m_rbm_conn */
265  m_rbm_conn.setZero();
266 
267  for (int i = 0; i < m_num_particles; i++)
268  {
269  const int i3{3 * i};
270 
271  const Eigen::Vector3d R_i = m_positions_particles.segment<3>(i3);
272 
273  // Calculate which body j particle i belongs to
274  const int body_j = (m_particle_type_id.segment(0, i + 1).array() == 1).count() - 1;
275  const int j6{6 * body_j};
276  const int j7{7 * body_j};
277 
278  // (negative) moment arm of particle about its body's locater position
279  Eigen::Vector3d n_dr = m_positions_bodies.segment<3>(j7);
280  n_dr.noalias() -= R_i;
281 
282  // skew-symmetric matrix representation of cross product
283  Eigen::Matrix3d n_dr_cross;
284  crossProdMat(n_dr, n_dr_cross);
285 
286  // rigid body motion connectivity tensor elements
287  m_rbm_conn.block<3, 3>(j6, i3).noalias() = m_I; // translation-translation couple
288  m_rbm_conn.block<3, 3>(j6 + 3, i3).noalias() = n_dr_cross; // translation-rotation couple
289 
290  if (i == 0)
291  {
292  assert(body_j == 0 && "First particle must be part of body 0.");
293  }
294  if (i == m_num_particles - 1)
295  {
296  assert(body_j + 1 == m_num_bodies && "Last particle must be part of body M.");
297  }
298  }
299 
300  /* ANCHOR: Compute m_psi_conv_quat_ang */
301  m_psi_conv_quat_ang.setZero();
302 
303  for (int k = 0; k < m_num_bodies; k++)
304  {
305  const int k6{6 * k};
306  const int k7{7 * k};
307 
308  // matrix E from quaterion of body k
309  Eigen::Matrix<double, 3, 4> E_theta_k;
310  eMatrix(m_positions_bodies.segment<4>(k7 + 3), E_theta_k);
311 
312  // matrix elements of Psi
313  m_psi_conv_quat_ang.block<3, 3>(k6, k7).noalias() =
314  m_I; // no conversion from linear components
315  m_psi_conv_quat_ang.block<3, 4>(k6 + 3, k7 + 3).noalias() =
316  2 * E_theta_k; // angular-quaternion velocity couple
317  }
318 
319  /* ANCHOR: Compute m_C_conv_quat_part */
320  m_C_conv_quat_part.noalias() = m_rbm_conn.transpose() * m_psi_conv_quat_ang;
321 }
