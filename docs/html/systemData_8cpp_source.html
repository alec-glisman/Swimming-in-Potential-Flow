174void
175systemData::velocitiesArticulation(double time)
176{
177 double dimensional_time{m_tau * time};
178
179 // ANCHOR: Orientation vectors, q = R_1 - R_3
180 Eigen::Vector3d q = m_positions_particles.segment<3>(3 * 0) - m_positions_particles.segment<3>(3 * 2);
181 q.normalize();
182 Eigen::Vector3d q_tilde = q;
183 q_tilde(2) *= -1;
184
185 // articulation velocity magnitudes
186 const double v1_mag = m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time);
187 const double v3_mag = m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time + m_sys_spec_phase_shift);
188
189 // Zero and then calculate m_velocities_particles_articulation
190 m_velocities_particles_articulation.setZero();
191 m_velocities_particles_articulation.segment<3>(3 * 0).noalias() = v1_mag * q;
192 m_velocities_particles_articulation.segment<3>(3 * 2).noalias() = v3_mag * q;
193 m_velocities_particles_articulation.segment<3>(3 * 3).noalias() = v1_mag * q_tilde;
194 m_velocities_particles_articulation.segment<3>(3 * 5).noalias() = v3_mag * q_tilde;
195}

256void
257systemData::rigidBodyMotionTensors()
258{
259 /* ANCHOR: Compute m_rbm_conn */
260 m_rbm_conn.setZero();
261
262 for (int i = 0; i < m_num_particles; i++)
263 {
264 const int i3{3 * i};
265 const Eigen::Vector3d R_i = m_positions_particles.segment<3>(i3);
266
267 // Calculate which body j particle i belongs to
268 const int body_j = (m_particle_type_id.segment(0, i + 1).array() == 1).count() - 1;
269 const int j6{6 * body_j};
270 const int j7{7 * body_j};
271
272 // (negative) moment arm of particle about its body's locater position
273 Eigen::Vector3d n_dr = m_positions_bodies.segment<3>(j7);
274 n_dr.noalias() -= R_i;
275
276 // skew-symmetric matrix representation of cross product
277 Eigen::Matrix3d n_dr_cross;
278 crossProdMat(n_dr, n_dr_cross);
279
280 // rigid body motion connectivity tensor elements
281 m_rbm_conn.block<3, 3>(j6, i3).noalias() = m_I; // translation-translation couple
282 m_rbm_conn.block<3, 3>(j6 + 3, i3).noalias() = n_dr_cross; // translation-rotation couple
283
284 if (i == 0)
285 {
286 assert(body_j == 0 && "First particle must be part of body 0.");
287 }
288 if (i == m_num_particles - 1)
289 {
290 assert(body_j + 1 == m_num_bodies && "Last particle must be part of body M.");
291 }
292 }
293
294 /* ANCHOR: Compute m_psi_conv_quat_ang */
295 m_psi_conv_quat_ang.setZero();
296
297 for (int k = 0; k < m_num_bodies; k++)
298 {
299 const int k6{6 * k};
300 const int k7{7 * k};
301
302 // matrix E from quaterion of body k
303 Eigen::Matrix<double, 3, 4> E_theta_k;
304 eMatrix(m_positions_bodies.segment<4>(k7 + 3), E_theta_k);
305
306 // matrix elements of Psi
307 m_psi_conv_quat_ang.block<3, 3>(k6, k7).noalias() = m_I; // no conversion from linear components
308 m_psi_conv_quat_ang.block<3, 4>(k6 + 3, k7 + 3).noalias() = 2 * E_theta_k; // angular-quaternion velocity couple
309 }
310
311 /* ANCHOR: Compute m_C_conv_quat_part */
312 m_C_conv_quat_part.noalias() = m_rbm_conn.transpose() * m_psi_conv_quat_ang;
313}

315void
316systemData::gradientChangeOfVariableTensors()
317{
319 Eigen::ThreadPoolDevice all_cores_device(&m_thread_pool, m_num_physical_cores);
320
321 /* ANCHOR: Compute m_D_conv_quat_part */
322 m_D_conv_quat_part.setZero();
323
324 int body_num{-1}; // -1 as first particle should be locater particle and increment this
325 Eigen::Matrix<double, 3, 4> twoE_body;
326 Eigen::Vector4d R_c_body;
327
328 assert(m_particle_type_id(0) == 1 && "First particle index must be locater by convention");
329 for (int j = 0; j < m_num_particles; j++)
330 {
331
332 if (m_particle_type_id(j) == 1)
333 {
334 // Zero out the relevent locater values
335 twoE_body.setZero();
336 R_c_body.setZero();
337 // Increment body count
338 body_num += 1;
339
340 // Get locater position of body i
341 R_c_body.segment<3>(1).noalias() = m_positions_locater_particles.segment<3>(3 * body_num);
342 // Get E matrix representation of body quaternion from m_psi_conv_quat_ang
343 twoE_body.noalias() = m_psi_conv_quat_ang.block<3, 4>(6 * body_num + 3, 7 * body_num + 3);
344
345 // Continue to next loop as all elements are zero
346 continue;
347 }
348
349 // j -> particle number
350 const int j3{3 * j};
351
352 // 4-vector version of particle j position (prepend zero element)
353 Eigen::Vector4d R_j = Eigen::Vector4d::Zero(4, 1);
354 R_j.segment<3>(1).noalias() = m_positions_particles.segment<3>(j3);
355
356 // matrix P_tilde from moment arm
357 const Eigen::Vector4d dr = R_j - R_c_body;
358 Eigen::Matrix<double, 3, 4> P_j_tilde_hold;
359 eMatrix(dr, P_j_tilde_hold);
360 Eigen::Matrix<double, 4, 3> P_j_tilde = P_j_tilde_hold.transpose();
361
362 // change of variables gradient tensor elements
363 m_D_conv_quat_part.block<3, 3>(7 * body_num, j3).noalias() = -m_I; // translation-translation couple
364 m_D_conv_quat_part.block<3, 3>(7 * body_num + 4, j3).noalias() =
365 twoE_body * P_j_tilde; // quaternion-rotation couple (first row is zero)
366 }
367 assert(body_num + 1 == m_num_bodies && "Not all bodies were indexed correctly");
368
369 /* ANCHOR : Compute m_C_conv_quat_part_grad */
370 m_C_conv_quat_part_grad.setZero();
371 body_num = -1;
372
373 assert(m_particle_type_id(0) == 1 && "First particle index must be locater by convention");
374 for (int i = 0; i < m_num_particles; i++)
375 {
376 if (m_particle_type_id(i) == 1)
377 {
378 // Increment body count
379 body_num += 1;
380
381 // Continue to next loop as all elements are zero
382 continue;
383 }
384
385 // get moment arm to locater point from C
386 const Eigen::Matrix3d two_r_cross_mat = -2 * m_rbm_conn.block<3, 3>(6 * body_num + 3, 3 * i);
387
388 // get E matrix representation of body quaternion from m_psi_conv_quat_ang
389 const Eigen::Matrix<double, 3, 4> two_E_body =
390 m_psi_conv_quat_ang.block<3, 4>(6 * body_num + 3, 7 * body_num + 3);
391
392 // get change of variable matrix element D_{\alpha}
393 const Eigen::Matrix<double, 7, 3> n_D_alpha = -m_D_conv_quat_part.block<7, 3>(7 * body_num, 3 * i);
394
395 /* ANCHOR: tensor contractions to produce result */
396
397 // convert Eigen::Matrix --> Eigen::Tensor
398 const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_two_r_cross_mat = TensorCast(two_r_cross_mat);
399 const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4>> tens_two_E_body = TensorCast(two_E_body);
400 const Eigen::TensorFixedSize<double, Eigen::Sizes<7, 3>> tens_n_D_alpha = TensorCast(n_D_alpha);
401
402 // 1) Compute result_{i m k} = levi_cevita{l i m} n_D_alpha_{k l}
403 Eigen::array<Eigen::IndexPair<int>, 1> contract_lim_kl = {Eigen::IndexPair<int>(0, 1)}; // {i m k}
404 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 7>> d_rCrossMat_d_xi;
405 d_rCrossMat_d_xi.device(all_cores_device) = m_levi_cevita.contract(tens_n_D_alpha, contract_lim_kl);
406
407 // 2) Compute result_{i j k} = d_rCrossMat_d_xi_{i m k} two_E_body{m j}
408 Eigen::array<Eigen::IndexPair<int>, 1> contract_imk_mj = {Eigen::IndexPair<int>(1, 0)}; // {i k j}, must shuffle
409 Eigen::array<int, 3> swap_last_two_indices({0, 2, 1});
410
411 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 7, 4>> d_rCrossMat_d_xi_times_two_E_body_preshuffle;
412 d_rCrossMat_d_xi_times_two_E_body_preshuffle.device(all_cores_device) =
413 d_rCrossMat_d_xi.contract(tens_two_E_body, contract_imk_mj);
414
415 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> d_rCrossMat_d_xi_times_two_E_body;
416 d_rCrossMat_d_xi_times_two_E_body.device(all_cores_device) =
417 d_rCrossMat_d_xi_times_two_E_body_preshuffle.shuffle(swap_last_two_indices);
418
419 // 3) Compute result_{i j k} = two_r_cross_mat_{i m} Kappa_tilde_{m j k} (same indices as (2))
420 Eigen::array<Eigen::IndexPair<int>, 1> contract_im_mjk = {Eigen::IndexPair<int>(1, 0)}; // {i j k}
421 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> rCrossMat_times_d_E_body_d_xi;
422 rCrossMat_times_d_E_body_d_xi.device(all_cores_device) =
423 tens_two_r_cross_mat.contract(m_kappa_tilde, contract_im_mjk);
424
425 // Compute and output matrix element
426 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> angular_gradient;
427 angular_gradient.device(all_cores_device) = -d_rCrossMat_d_xi_times_two_E_body - rCrossMat_times_d_E_body_d_xi;
428
429 // output matrix indices
430 Eigen::array<Eigen::Index, 3> offsets = {3 * i, 7 * body_num + 3, 7 * body_num};
431 Eigen::array<Eigen::Index, 3> extents = {3, 4, 7};
432 m_C_conv_quat_part_grad.slice(offsets, extents) = angular_gradient;
433 }
434 assert(body_num + 1 == m_num_bodies && "Not all bodies were indexed correctly");
435}
