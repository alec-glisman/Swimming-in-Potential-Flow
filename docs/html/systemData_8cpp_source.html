134void
135systemData::velocitiesArticulation(double time)
136{
137 double dimensional_time{m_tau * time};
138
139 // ANCHOR: Orientation vectors, q = R_1 - R_3
140 Eigen::Vector3d q =
141 m_positions_particles.segment<3>(3 * 0) - m_positions_particles.segment<3>(3 * 2);
142 q.normalize();
143 Eigen::Vector3d q_tilde = m_I_tilde * q;
144
145 // articulation velocity magnitudes
146 const double v1_mag = m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time);
147 const double v3_mag =
148 m_sys_spec_U0 * cos(m_sys_spec_omega * dimensional_time + m_sys_spec_phase_shift);
149
150 // Zero and then calculate m_velocities_particles_articulation
151 m_velocities_particles_articulation = Eigen::VectorXd::Zero(3 * m_num_particles);
152 m_velocities_particles_articulation.segment<3>(3 * 0).noalias() = v1_mag * q;
153 m_velocities_particles_articulation.segment<3>(3 * 2).noalias() = v3_mag * q;
154 m_velocities_particles_articulation.segment<3>(3 * 3).noalias() = v1_mag * q_tilde;
155 m_velocities_particles_articulation.segment<3>(3 * 5).noalias() = v3_mag * q_tilde;
156}

158/* REVIEW[epic=Change,order=4]: Change assignment of m_accelerations_particles_articulation for
159 * different systems */
160void
161systemData::accelerationsArticulation(double time)
162{
163 double dimensional_time{m_tau * time};
164
165 // ANCHOR: Orientation vectors, q = R_1 - R_3
166 Eigen::Vector3d q =
167 m_positions_particles.segment<3>(3 * 0) - m_positions_particles.segment<3>(3 * 2);
168 q.normalize();
169 Eigen::Vector3d q_tilde = m_I_tilde * q;
170
171 // articulation acceleration magnitudes
172 const double a1_mag =
173 -m_sys_spec_U0 * m_sys_spec_omega * sin(m_sys_spec_omega * dimensional_time);
174 const double a3_mag = -m_sys_spec_U0 * m_sys_spec_omega *
175 sin(m_sys_spec_omega * dimensional_time + m_sys_spec_phase_shift);
176
177 // Zero and then calculate m_accelerations_particles_articulation
178 m_accelerations_particles_articulation = Eigen::VectorXd::Zero(3 * m_num_particles);
179 m_accelerations_particles_articulation.segment<3>(3 * 0).noalias() = a1_mag * q;
180 m_accelerations_particles_articulation.segment<3>(3 * 2).noalias() = a3_mag * q;
181 m_accelerations_particles_articulation.segment<3>(3 * 3).noalias() = a1_mag * q_tilde;
182 m_accelerations_particles_articulation.segment<3>(3 * 5).noalias() = a3_mag * q_tilde;
183}

189{

197 {
198 m_positions_bodies.segment<3>(3 * body_count).noalias() =
199 m_positions_particles.segment<3>(3 * i);
200
201 body_count += 1;
202 }
203 }
204 assert(body_count == m_num_bodies && "Incorrect number of bodies filled");
205}
