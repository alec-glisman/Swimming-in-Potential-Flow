362 void
363 systemData::positionsArticulation()
364 {
365  double t_dimensional{m_tau * m_t};
366 
367  // articulation acceleration magnitudes
368  const double r1_mag = m_sys_spec_R_avg + (m_sys_spec_U0 / m_sys_spec_omega) * sin(m_sys_spec_omega * t_dimensional);
369  const double r3_mag = m_sys_spec_R_avg - (m_sys_spec_U0 / m_sys_spec_omega) *
370  sin(m_sys_spec_omega * t_dimensional + m_sys_spec_phase_shift);
371 
372  Eigen::VectorXd particle_distances(m_num_particles);
373  particle_distances << 0.0, r1_mag, r3_mag, 0.0, r1_mag, r3_mag;
374 
375  m_positions_particles_articulation.setZero();
376 
377  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
378  {
379  if (m_particle_type_id(particle_id) == 1)
380  {
381  continue; // continue to next loop as all elements are zero
382  }
383 
384  const int particle_id_3{3 * particle_id};
385  const int body_id_7{7 * m_particle_group_id(particle_id)};
386 
387  m_positions_particles_articulation.segment<3>(particle_id_3).noalias() =
388  particle_distances(particle_id) * m_orientations_particles.segment<3>(particle_id_3);
389  }

481 systemData::rigidBodyMotionTensors(Eigen::ThreadPoolDevice& device)
482 {
483  /* ANCHOR: Compute m_rbm_conn */
484  m_rbm_conn.setZero();
485 
486  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
487  {
488  if (m_particle_type_id(particle_id) == 1)
489  {
490  continue; // continue to next loop as all elements are zero
491  }
492 
493  const int particle_id_3{3 * particle_id};
494  const int body_id_6{6 * m_particle_group_id(particle_id)};
495 
496  // skew-symmetric matrix representation of (negative) cross product
497  Eigen::Matrix3d n_dr_cross;
498  crossProdMat(-m_positions_particles_articulation.segment<3>(particle_id_3), n_dr_cross);
499 
500  // rigid body motion connectivity tensor elements
501  m_rbm_conn.block<3, 3>(body_id_6, particle_id_3).noalias() = m_I3; // translation-translation couple
502  m_rbm_conn.block<3, 3>(body_id_6 + 3, particle_id_3).noalias() = n_dr_cross; // translation-rotation couple
503  }
504 
505  /* ANCHOR: Compute m_psi_conv_quat_ang */
506  m_psi_conv_quat_ang.setZero();
507 
508  for (int body_id = 0; body_id < m_num_bodies; body_id++)
509  {
510  const int body_id_6{6 * body_id};
511  const int body_id_7{7 * body_id};
512 
513  // matrix E from quaterion of body k
514  Eigen::Matrix<double, 3, 4> E_theta_k;
515  eMatrix(m_positions_bodies.segment<4>(body_id_7 + 3), E_theta_k);
516 
517  // matrix elements of Psi
518  m_psi_conv_quat_ang.block<3, 3>(body_id_6, body_id_7).noalias() = m_I3; // no conversion from linear components
519  m_psi_conv_quat_ang.block<3, 4>(body_id_6 + 3, body_id_7 + 3).noalias() =
520  2 * E_theta_k; // angular-quaternion velocity couple
521  }
522 
523  /* ANCHOR: Compute m_rbm_conn_T_quat & m_tens_rbm_conn_T_quat */
524  m_rbm_conn_T_quat.noalias() = m_rbm_conn.transpose() * m_psi_conv_quat_ang;
525  m_tens_rbm_conn_T_quat = TensorCast(m_rbm_conn_T_quat);
526 }

528 void
529 systemData::gradientChangeOfVariableTensors(Eigen::ThreadPoolDevice& device)
530 {
531  /* ANCHOR: Compute m_conv_body_2_part_dof and m_tens_conv_body_2_part_dof */
532  m_conv_body_2_part_dof.setZero();
533 
534  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
535  {
536  if (m_particle_type_id(particle_id) == 1)
537  {
538  continue; // continue to next loop as all elements are zero
539  }
540 
541  const int particle_id_3{3 * particle_id};
542  const int body_id_6{6 * m_particle_group_id(particle_id)};
543  const int body_id_7{7 * m_particle_group_id(particle_id)};
544 
545  // (4 vector) displacement of particle from locater point moment arm
546  Eigen::Vector4d dr = Eigen::Vector4d::Zero(4, 1);
547  dr.segment<3>(1).noalias() = m_positions_particles_articulation.segment<3>(particle_id_3);
548 
549  // quaternion product representation of particle displacement (transpose)
550  Eigen::Matrix<double, 3, 4> P_j_tilde_T;
551  eMatrix(dr, P_j_tilde_T);
552  Eigen::Matrix<double, 4, 3> P_j_tilde = P_j_tilde_T.transpose();
553 
554  // change of variables gradient tensor elements
555  m_conv_body_2_part_dof.block<3, 3>(body_id_7, particle_id_3).noalias() =
556  -m_I3; // translation-translation couple
557  m_conv_body_2_part_dof.block<3, 3>(body_id_7 + 4, particle_id_3).noalias() =
558  m_psi_conv_quat_ang.block<3, 4>(body_id_6 + 3, body_id_7 + 3) *
559  P_j_tilde; // quaternion-rotation couple (first row is zero)
560  }
561 
562  m_tens_conv_body_2_part_dof = TensorCast(m_conv_body_2_part_dof);
563 
564  /* ANCHOR : Compute m_rbm_conn_T_quat_grad */
565  m_rbm_conn_T_quat_grad.setZero();
566 
567  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
568  {
569  if (m_particle_type_id(particle_id) == 1)
570  {
571  // Continue to next loop as all elements are zero
572  continue;
573  }
574 
575  const int particle_id_3{3 * particle_id};
576  const int body_id_6{6 * m_particle_group_id(particle_id)};
577  const int body_id_7{7 * m_particle_group_id(particle_id)};
578 
579  // get moment arm to locater point from C
580  const Eigen::Matrix3d two_r_cross_mat = -2 * m_rbm_conn.block<3, 3>(body_id_6 + 3, particle_id_3);
581 
582  // get E matrix representation of body quaternion from m_psi_conv_quat_ang
583  const Eigen::Matrix<double, 3, 4> two_E_body = m_psi_conv_quat_ang.block<3, 4>(body_id_6 + 3, body_id_7 + 3);
584 
585  // get change of variable matrix element D_{\alpha}
586  const Eigen::Matrix<double, 7, 3> n_D_alpha = -m_conv_body_2_part_dof.block<7, 3>(body_id_7, particle_id_3);
587 
588  /* ANCHOR: tensor contractions to produce result */
589 
590  // convert Eigen::Matrix --> Eigen::Tensor
591  const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_two_r_cross_mat = TensorCast(two_r_cross_mat);
592  const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4>> tens_two_E_body = TensorCast(two_E_body);
593  const Eigen::TensorFixedSize<double, Eigen::Sizes<7, 3>> tens_n_D_alpha = TensorCast(n_D_alpha);
594 
595  // 1) Compute result_{i m k} = levi_cevita{l i m} n_D_alpha_{k l}
596  const Eigen::array<Eigen::IndexPair<int>, 1> contract_lim_kl = {Eigen::IndexPair<int>(0, 1)}; // {i m k}
597  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 7>> d_rCrossMat_d_xi;
598  d_rCrossMat_d_xi.device(device) = m_levi_cevita.contract(tens_n_D_alpha, contract_lim_kl);
599 
600  // 2) Compute result_{i j k} = d_rCrossMat_d_xi_{i m k} two_E_body{m j}
601  const Eigen::array<Eigen::IndexPair<int>, 1> contract_imk_mj = {
602  Eigen::IndexPair<int>(1, 0)}; // {i k j}, must shuffle
603  const Eigen::array<int, 3> swap_last_two_indices({0, 2, 1});
604 
605  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 7, 4>> d_rCrossMat_d_xi_times_two_E_body_preshuffle;
606  d_rCrossMat_d_xi_times_two_E_body_preshuffle.device(device) =
607  d_rCrossMat_d_xi.contract(tens_two_E_body, contract_imk_mj);
608 
609  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> d_rCrossMat_d_xi_times_two_E_body;
610  d_rCrossMat_d_xi_times_two_E_body.device(device) =
611  d_rCrossMat_d_xi_times_two_E_body_preshuffle.shuffle(swap_last_two_indices);
612 
613  // 3) Compute result_{i j k} = two_r_cross_mat_{i m} Kappa_tilde_{m j k} (same indices as (2))
614  const Eigen::array<Eigen::IndexPair<int>, 1> contract_im_mjk = {Eigen::IndexPair<int>(1, 0)}; // {i j k}
615  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> rCrossMat_times_d_E_body_d_xi;
616  rCrossMat_times_d_E_body_d_xi.device(device) = tens_two_r_cross_mat.contract(m_kappa_tilde, contract_im_mjk);
617 
618  // Compute and output matrix element
619  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4, 7>> angular_gradient;
620  angular_gradient.device(device) = -d_rCrossMat_d_xi_times_two_E_body - rCrossMat_times_d_E_body_d_xi;
621 
622  // output matrix indices
623  const Eigen::array<Eigen::Index, 3> offsets = {particle_id_3, body_id_7 + 3, body_id_7};
624  const Eigen::array<Eigen::Index, 3> extents = {3, 4, 7};
625  m_rbm_conn_T_quat_grad.slice(offsets, extents) = angular_gradient;
626  }
