384 void
385 SystemData::positionsArticulation()
386 {
387  double t_dimensional{m_tau * m_t};
388 
389  // articulation acceleration magnitudes
390  const double r1_mag = m_sys_spec_R_avg + (m_sys_spec_U0 / m_sys_spec_omega) * sin(m_sys_spec_omega * t_dimensional);
391  const double r3_mag = m_sys_spec_R_avg - (m_sys_spec_U0 / m_sys_spec_omega) *
392  sin(m_sys_spec_omega * t_dimensional + m_sys_spec_phase_shift);
393 
394  Eigen::VectorXd particle_distances(m_num_particles);
395  particle_distances << 0.0, r1_mag, r3_mag, 0.0, r1_mag, r3_mag;
396 
397  m_positions_particles_articulation.setZero();
398 
399  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
400  {
401  if (m_particle_type_id(particle_id) == 1)
402  {
403  continue; // continue to next loop as all elements are zero
404  }
405 
406  const int particle_id_3{3 * particle_id};
407  const int body_id_7{7 * m_particle_group_id(particle_id)};
408 
409  m_positions_particles_articulation.segment<3>(particle_id_3).noalias() =
410  particle_distances(particle_id) * m_orientations_particles.segment<3>(particle_id_3);
411  }

532  }

538  {

540 
541  // matrix E from quaterion of body k

543  eMatrix(m_positions_bodies.segment<4>(body_id_7 + 3), E_theta_k);

547  m_psi_conv_quat_ang.block<4, 4>(body_id_7 + 3, body_id_7 + 3).noalias() =
548  2 * E_theta_k; // angular-quaternion velocity couple
549  }

556 void
557 SystemData::gradientChangeOfVariableTensors(Eigen::ThreadPoolDevice& device)
558 {
559  /* ANCHOR: Compute m_chi and m_tens_chi */
560  m_chi.setZero();
561 
562  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
563  {
564  const int particle_id_3{3 * particle_id};
565  const int particle_id_7{7 * particle_id};
566  const int body_id_7{7 * m_particle_group_id(particle_id)};
567 
568  const bool is_locater{m_particle_group_id(particle_id) == 1};
569 
571  const int s_part{-1 + 2 * is_locater};
572 
573  // (4 vector) displacement of particle from locater point moment arm
574  Eigen::Vector4d dr_init = Eigen::Vector4d::Zero(4, 1);
575  dr_init.segment<3>(1).noalias() = m_positions_particles_articulation_init_norm.segment<3>(particle_id_3);
576 
577  const double r_alpha_norm{m_positions_particles_articulation.segment<3>(particle_id_3).norm()};
578  dr_init *= r_alpha_norm; // incorporate norm factor here rather than in chi calculation for numerical efficiency
579 
580  // quaternion product representation of particle displacement (transpose)
581  Eigen::Matrix<double, 4, 4> g_T;
582  eMatrix(dr_init, g_T);
583 
584  // change of variables gradient tensor elements
585  m_chi.block<3, 3>(body_id_7, particle_id_7).noalias() = s_part * m_I3; // translation-translation couple
586  m_chi.block<4, 4>(body_id_7 + 4, particle_id_7).noalias() =
587  m_psi_conv_quat_ang.block<4, 4>(body_id_7 + 3, body_id_7 + 3) *
588  g_T.transpose(); // quaternion-rotation couple (first row is zero)
589  }
590 
591  m_tens_chi = TensorCast(m_chi);
592 
593  /* ANCHOR : Compute m_tens_grad_zeta */
594  m_tens_grad_zeta.setZero();
595 
596  for (int particle_id = 0; particle_id < m_num_particles; particle_id++)
597  {
598  /* ANCHOR: Tensor indices */
599  const int particle_id_7{7 * particle_id};
600  const int body_id_7{7 * m_particle_group_id(particle_id)};
601 
602  const bool is_locater{m_particle_group_id(particle_id) == 1};
603 
604  // `Eigen::Tensor` contract indices
605  const Eigen::array<Eigen::IndexPair<int>, 1> contract_ilk_lj = {
606  Eigen::IndexPair<int>(1, 0)}; // {i, l, k} . {l, j} --> {i, k, j}
607  const Eigen::array<Eigen::IndexPair<int>, 1> contract_li_ljk = {
608  Eigen::IndexPair<int>(0, 0)}; // {l, i} . {l, j, k} --> {i, j, k}
609  const Eigen::array<Eigen::IndexPair<int>, 1> contract_ljm_km = {
610  Eigen::IndexPair<int>(2, 1)}; // {l, j, m} . {k, m} --> {l, j, k}
611 
612  // `Eigen::Tensor` permute indices
613  const Eigen::array<int, 3> permute_ikj_ijk({0, 2, 1}); // {i, k, j} --> {i, j, k}
614 
615  // `Eigen::Tensor` output tensor indices start location (offset) and extent
616  const Eigen::array<Eigen::Index, 3> offsets_left = {body_id_7 + 3, particle_id_7, body_id_7};
617  const Eigen::array<Eigen::Index, 3> offsets_right = {body_id_7 + 3, particle_id_7 + 3, body_id_7};
618  const Eigen::array<Eigen::Index, 3> extents = {4, 3, 7};
619 
620  // get moment arm to locater point from particle
621  const Eigen::Matrix3d two_r_cross_mat = 2 * m_rbm_conn.block<3, 3>(body_id_7 + 3, particle_id_7);
622  const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_two_r_cross_mat = TensorCast(two_r_cross_mat);
623 
624  // get E matrix representation of body quaternion from m_psi_conv_quat_ang
625  const Eigen::Matrix<double, 3, 4> two_E_body = m_psi_conv_quat_ang.block<3, 4>(body_id_7 + 3, body_id_7 + 3);
626  const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 4>> tens_two_E_body = TensorCast(two_E_body);
627 
628  // get change of variable matrix element m_chi_tilde
629  const Eigen::Matrix<double, 7, 3> n_chi_tilde = -m_chi.block<7, 3>(body_id_7, particle_id_7);
630  const Eigen::TensorFixedSize<double, Eigen::Sizes<7, 3>> tens_n_chi_tilde = TensorCast(n_chi_tilde);
631 
632  /* ANCHOR: tensor contractions for left-half of gradient */
633  // compute grad_r_cross{l, j, k} = - m_levi_cevita{l, j, m} chi_tilde{k, m}
634  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 7>> grad_r_cross; // (3, 3, 7) {l, j, k}
635  grad_r_cross.device(device) = m_levi_cevita.contract(tens_n_chi_tilde, contract_ljm_km);
636 
637  // compute 2E_T_grad_r{i, j, k} = 2E_{l, i} grad_r_cross{l, j, k}
638  Eigen::TensorFixedSize<double, Eigen::Sizes<4, 3, 7>> two_E_grad_r_cross; // (4, 3, 7) {i, j, k}
639  two_E_grad_r_cross.device(device) = tens_two_E_body.contract(grad_r_cross, contract_li_ljk);
640 
641  // compute 2grad_E_r_cross_{i, k, j} = Kappa{i, l, k} r_cross{l, j}
642  Eigen::TensorFixedSize<double, Eigen::Sizes<4, 7, 3>> two_grad_E_r_cross_preshuffle; // (4, 3, 7) {i, k, j}
643  two_grad_E_r_cross_preshuffle.device(device) = m_kappa.contract(tens_two_r_cross_mat, contract_ilk_lj);
644 
645  // shuffle two_grad_E_r_cross_preshuffle
646  Eigen::TensorFixedSize<double, Eigen::Sizes<4, 3, 7>> two_grad_E_r_cross; // (4, 3, 7) {i, j, k}
647  two_grad_E_r_cross.device(device) = two_grad_E_r_cross_preshuffle.shuffle(permute_ikj_ijk);
648 
649  // left term
650  const Eigen::TensorFixedSize<double, Eigen::Sizes<4, 3, 7>> mixed_gradient =
651  two_grad_E_r_cross + two_E_grad_r_cross;
652  m_tens_grad_zeta.slice(offsets_left, extents) = mixed_gradient;
653 
654  /* ANCHOR: tensor contractions for right-half of gradient */
655  m_tens_grad_zeta.slice(offsets_right, extents) = 2 * m_kappa;
656  }
