171 inline void StopWatchWin::stop() {
172  QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&end_time));
173  diff_time = static_cast<float>(((static_cast<double>(end_time.QuadPart) -
174  static_cast<double>(start_time.QuadPart)) /
175  freq));
176 
177  total_time += diff_time;
178  clock_sessions++;
179  running = false;
180 }
181 
186 inline void StopWatchWin::reset() {
187  diff_time = 0;
188  total_time = 0;
189  clock_sessions = 0;
190 
191  if (running) {
192  QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&start_time));
193  }
194 }

202 inline float StopWatchWin::getTime() {
203  // Return the TOTAL time to date
204  float retval = total_time;
205 
206  if (running) {
207  LARGE_INTEGER temp;
208  QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&temp));
209  retval += static_cast<float>(((static_cast<double>(temp.QuadPart) -
210  static_cast<double>(start_time.QuadPart)) /
211  freq));
212  }
213 
214  return retval;
215 }

351 inline float StopWatchLinux::getDiffTime() {
352  struct timeval t_time;
353  gettimeofday(&t_time, 0);
354 
355  // time difference in milli-seconds
356  return static_cast<float>(1000.0 * (t_time.tv_sec - start_time.tv_sec) +
357  (0.001 * (t_time.tv_usec - start_time.tv_usec)));
358 }
