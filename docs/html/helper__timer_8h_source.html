80class StopWatchWin : public StopWatchInterface {
81 public:
83 StopWatchWin()
84 : start_time(),
85 end_time(),
86 diff_time(0.0f),
87 total_time(0.0f),
88 running(false),
89 clock_sessions(0),
90 freq(0),
91 freq_set(false) {
92 if (!freq_set) {
93 // helper variable
94 LARGE_INTEGER temp;
95
96 // get the tick frequency from the OS
97 QueryPerformanceFrequency(reinterpret_cast<LARGE_INTEGER *>(&temp));
98
99 // convert to type in which it is needed

101
102 // rememeber query

104 }
105 }

162inline void StopWatchWin::start() {
163 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&start_time));
164 running = true;
165}
166
171inline void StopWatchWin::stop() {
172 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&end_time));
173 diff_time = static_cast<float>(((static_cast<double>(end_time.QuadPart) -
174 static_cast<double>(start_time.QuadPart)) /
175 freq));
176
177 total_time += diff_time;
178 clock_sessions++;
179 running = false;
180}
181
186inline void StopWatchWin::reset() {
187 diff_time = 0;
188 total_time = 0;
189 clock_sessions = 0;
190
191 if (running) {
192 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&start_time));
193 }
194}

202inline float StopWatchWin::getTime() {
203 // Return the TOTAL time to date
204 float retval = total_time;
205
206 if (running) {
207 LARGE_INTEGER temp;
208 QueryPerformanceCounter(reinterpret_cast<LARGE_INTEGER *>(&temp));
209 retval += static_cast<float>(((static_cast<double>(temp.QuadPart) -
210 static_cast<double>(start_time.QuadPart)) /
211 freq));
212 }
213
214 return retval;
215}
216
221inline float StopWatchWin::getAverageTime() {
222 return (clock_sessions > 0) ? (total_time / clock_sessions) : 0.0f;
223}

351inline float StopWatchLinux::getDiffTime() {
352 struct timeval t_time;
353 gettimeofday(&t_time, 0);
354
355 // time difference in milli-seconds
356 return static_cast<float>(1000.0 * (t_time.tv_sec - start_time.tv_sec) +
357 (0.001 * (t_time.tv_usec - start_time.tv_usec)));
358}
