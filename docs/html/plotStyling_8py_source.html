520# @Source: https://stackoverflow.com/questions/40642061/how-to-set-axis-ticks-in-multiples-of-pi-python-matplotlib
521# @User: Scott Centoni
522def multiple_formatter(denominator=2, number=np.pi, latex=r'\pi'):
523
524 def gcd(a, b):
525
526 while b:
527 a, b = b, a % b
528
529 return a
530
531 def _multiple_formatter(x, pos):
532
533 den = denominator
534 num = np.int(np.rint(den*x/number))
535 com = gcd(num, den)
536 (num, den) = (int(num/com), int(den/com))
537
538 if den == 1:
539
540 if num == 0:
541 return r'$0$'
542
543 if num == 1:
544 return r'$%s$' % latex
545
546 elif num == -1:
547 return r'$-%s$' % latex
548
549 else:
550 return r'$%s%s$' % (num, latex)
551 else:
552
553 if num == 1:
554 return r'$\frac{%s}{%s}$' % (latex, den)
555
556 elif num == -1:
557 return r'$\frac{-%s}{%s}$' % (latex, den)
558
559 else:
560 return r'$\frac{%s%s}{%s}$' % (num, latex, den)
561
562 return _multiple_formatter
