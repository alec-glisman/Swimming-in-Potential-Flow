44// Evaluates tensor expressions if needed
45template <typename T, typename Device = Eigen::DefaultDevice>
46auto
47asEval(const Eigen::TensorBase<T, Eigen::ReadOnlyAccessors>&
48 expr, // An Eigen::TensorBase object (Tensor, TensorMap, TensorExpr... )
49 const Device& device = Device() // Override to evaluate on another device, e.g. thread pool or gpu.
50)
51{
52 using Evaluator = Eigen::TensorEvaluator<const Eigen::TensorForcedEvalOp<const T>, Device>;
53 Eigen::TensorForcedEvalOp<const T> eval = expr.eval();
54 Evaluator tensor(eval, device);
55 tensor.evalSubExprsIfNeeded(nullptr);
56 return tensor;
57}

59// Converts any Eigen::Tensor (or expression) to an Eigen::Matrix with shape rows/cols
60template <typename T, typename sizeType, typename Device = Eigen::DefaultDevice>
61auto
62MatrixCast(const Eigen::TensorBase<T, Eigen::ReadOnlyAccessors>& expr, const sizeType rows, const sizeType cols,
63 const Device& device = Device())
64{
65 auto tensor = asEval(expr, device);
66 using Scalar = typename Eigen::internal::remove_const<typename decltype(tensor)::Scalar>::type;
67 return static_cast<MatrixType<Scalar>>(Eigen::Map<const MatrixType<Scalar>>(tensor.data(), rows, cols));
68}

70// Converts any Eigen::Tensor (or expression) to an Eigen::Vector with the same size
71template <typename T, typename Device = Eigen::DefaultDevice>
72auto
73VectorCast(const Eigen::TensorBase<T, Eigen::ReadOnlyAccessors>& expr, const Device& device = Device())
74{
75 auto tensor = asEval(expr, device);
76 auto size = Eigen::internal::array_prod(tensor.dimensions());
77 using Scalar = typename Eigen::internal::remove_const<typename decltype(tensor)::Scalar>::type;
78 return static_cast<VectorType<Scalar>>(Eigen::Map<const VectorType<Scalar>>(tensor.data(), size));
79}

143// Converts an Eigen::Matrix (or expression) to Eigen::Tensor
144// with dimensions directly as arguments in a variadic template
145template <typename Derived>
146auto
147TensorCast(const Eigen::EigenBase<Derived>& matrix)
148{
149 if constexpr (Derived::ColsAtCompileTime == 1 or Derived::RowsAtCompileTime == 1)
