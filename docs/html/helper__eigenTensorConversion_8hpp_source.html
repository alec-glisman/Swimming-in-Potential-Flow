54 template <typename Derived, typename Device = Eigen::DefaultDevice> class selfCleaningEvaluator
55 {
56  private:
57  using Evaluator = Eigen::TensorEvaluator<const Eigen::TensorForcedEvalOp<const Derived>, Device>;
58  Evaluator m_eval;
59 
60  public:
61  selfCleaningEvaluator(const Evaluator& eval) : m_eval(eval)
62  {
63  }
64  selfCleaningEvaluator(const Eigen::TensorBase<Derived, Eigen::ReadOnlyAccessors>& expr,
65  const Device& device = Device())
66  : m_eval(Evaluator(expr.eval(), device))
67  {
68  m_eval.evalSubExprsIfNeeded(nullptr);
69  }

71  ~selfCleaningEvaluator()
72  {
73  // This whole point of this object is to call cleanup automatically on destruct.
74  // If there are pending operations to evaluate, m_eval will allocate a buffer to hold a result,
75  // which needs to be deallocated.
76  m_eval.cleanup();
77  }

80  rank()
81  {
82  using DimType = typename decltype(m_eval)::Dimensions::Base;
83  return DimType{}.size(); // Because Derived::Dimensions is sometimes wrong
84  }
85  const Evaluator*
86  operator->() const
87  {
88  return &m_eval;
89  }
90  Evaluator*
91  operator->()
92  {
93  return &m_eval;

95  constexpr auto
96  map()
97  {
98  // We inspect m_eval to get the type, rank and dimensions, because it has the resulting tensor,
99  // whereas Derived is the tensor type _before_ whatever operation is pending (if any).
100  using DimType = typename decltype(m_eval)::Dimensions::Base;
101  constexpr auto rank = DimType{}.size();
102  using Scalar = typename Eigen::internal::remove_const<typename decltype(m_eval)::Scalar>::type;
103  return Eigen::TensorMap<Eigen::Tensor<Scalar, rank>>(m_eval.data(), m_eval.dimensions());
104  }
105 };

117 // Converts any Eigen::Tensor (or expression) to an Eigen::Matrix with shape rows/cols
118 template <typename T, typename sizeType, typename Device = Eigen::DefaultDevice>
119 auto
120 MatrixCast(const Eigen::TensorBase<T, Eigen::ReadOnlyAccessors>& expr, const sizeType rows, const sizeType cols,
121  const Device& device = Device())
122 {
123  auto tensor = asEval(expr, device);
124  auto tensorMap = tensor.map();
125  using Scalar = typename decltype(tensorMap)::Scalar;
126  return static_cast<MatrixType<Scalar>>(Eigen::Map<const MatrixType<Scalar>>(tensorMap.data(), rows, cols));
127 }

129 // Converts any Eigen::Tensor (or expression) to an Eigen::Matrix with shape rows/cols
130 template <typename T, typename Device = Eigen::DefaultDevice>
131 auto
132 VectorCast(const Eigen::TensorBase<T, Eigen::ReadOnlyAccessors>& expr, const Device& device = Device())
133 {
134  auto tensor = asEval(expr, device);
135  auto tensorMap = tensor.map();
136  auto size = Eigen::internal::array_prod(tensorMap.dimensions());
137  using Scalar = typename decltype(tensorMap)::Scalar;
138  return static_cast<VectorType<Scalar>>(Eigen::Map<const VectorType<Scalar>>(tensorMap.data(), size));
139 }
