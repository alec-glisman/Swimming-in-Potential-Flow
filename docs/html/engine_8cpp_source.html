46void
47engine::run()
48{
49 spdlog::get(m_logName)->critical("Starting engine run");
50 m_progressBar->display(); // display the progress bar
51
52 // calculate number of steps in integration
53 double t_remaining{m_system->tf() - m_system->t()};
54
55 int tot_step = (int)ceil(t_remaining / m_system->dt());
56 int write_step = (int)ceil(t_remaining / m_system->dt() / m_system->numStepsOutput());
57 int display_step = (int)ceil(t_remaining / m_system->dt() * m_outputPercentile);
58
59 spdlog::get(m_logName)->info("Write step: {0}", write_step);
60 spdlog::get(m_logName)->info("Display step: {0}", display_step);
61
62 // create eigen3 thread-pool and device
64 Eigen::ThreadPool thread_pool = Eigen::ThreadPool(m_num_physical_cores);
66 Eigen::ThreadPoolDevice all_cores_device(&thread_pool, m_num_physical_cores);
67
68 while (m_system->timestep() < tot_step)
69 {
70 // Integrate system forward in time
71 integrate(all_cores_device);
72
73 // Update time for next step
74 m_system->setT(m_system->t() + m_system->dt());
75 m_system->setTimestep(m_system->timestep() + 1);
76 ++(*m_progressBar);
77
78 // Output data
79 if ((m_system->timestep() % write_step == 0) || (m_system->t() >= m_system->tf()))
80 {
81 spdlog::get(m_logName)->info("Writing frame at t = {0}", m_system->t());
82 m_system->gsdUtil()->writeFrame();
83 spdlog::get(m_logName)->flush();
84 }
85 if (m_system->timestep() % display_step == 0)
86 {
87 m_progressBar->display(); // display the progress bar
88 }
89 }
90
91 // Final data writing and shut down
92 spdlog::get(m_logName)->info("Ending engine run");
93 spdlog::get(m_logName)->info("Writing frame at t = {0}", m_system->t());
94 m_system->gsdUtil()->writeFrame();
95 m_progressBar->display();
96 spdlog::get(m_logName)->flush();
97}
