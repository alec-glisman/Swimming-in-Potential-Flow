41 def f_D(x):
42  """Function for D: derivative of mass matrix elements, dimensionless
43 
44  Args:
45  x (np.array): Numpy array of average particle pair separation
46 
47  Returns:
48  np.array: Leading order gradient term in Taylor expansion for locater point net translation after one period of articulation.
49  """
50 
51  # In R/a coordinates
52  numerator = 3.0 * (-68.0 + 93.0 * np.power(x, 3))
53  denominator = x * np.square(17.0 - 18.0 * np.power(x, 3))
54  return np.divide(numerator, denominator)

58 def dZ_leadingOrder(phi, U0, omega, a, x):
59  """Calculates leading order net translation of locater point for collinear swimmer after one period of articulation
60 
61  Args:
62  phi (float): phase_angle between oscillator pairs
63  U0 (float): velocity articulation oscillation amplitude
64  omega (float): oscillation frequency
65  a (float): radius of spheres
66  x (float): average particle pair separation
67 
68  Returns:
69  float: Leading order Taylor expansion for locater point net translation after one period of articulation
70  """
71 
72  # In R/a coordinates
73  return -(np.pi * np.sin(phi)) * np.square(U0 / (a * omega)) * f_D(x)

78  """Calculates the relative error of approximate relative to exact solutions

80  Args:
81  exact (float): Exact solution to use as ground truth
82  approximate (float): Approximate numerical result

84  Returns:
