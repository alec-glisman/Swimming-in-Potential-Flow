46 rungeKutta4::integrateSecondOrder(Eigen::ThreadPoolDevice& device)
47 {
48  /* ANCHOR: Solve system of form: y'(t) = f( y(t), t )
49  * @REFERENCE:
50  * https://www.physicsforums.com/threads/using-runge-kutta-method-for-position-calc.553663/post-3634957
51  */
52 
53  /* Step 1: k1 = f( y(t_0), t_0 ),
54  * initial conditions at current step */
55  const double t1{m_system->t()};
56  const Eigen::VectorXd v1 = m_system->velocitiesParticles();
57  const Eigen::VectorXd x1 = m_system->positionsParticles();
58  Eigen::VectorXd a1 = Eigen::VectorXd::Zero(3 * m_system->numParticles());
59  accelerationUpdate(a1, device);
60 
61  /* Step 2: k2 = f( y(t_0) + k1 * dt/2, t_0 + dt/2 )
62  * time rate-of-change k1 evaluated halfway through time step (midpoint) */
63  Eigen::VectorXd v2 = v1;
64  v2.noalias() += m_c1_2_dt * a1;
65  Eigen::VectorXd x2 = x1;
66  x2.noalias() += m_c1_2_dt * v2;
67 
68  m_system->setVelocitiesParticles(v2);
69  m_system->setPositionsParticles(x2);
70 
71  m_system->setT(t1 + 0.50 * m_system->dt());
72 
73  Eigen::VectorXd a2 = Eigen::VectorXd::Zero(3 * m_system->numParticles());
74  accelerationUpdate(a2, device);
75 
76  /* Step 3: k3 = f( y(t_0) + k2 * dt/2, t_0 + dt/2 )
77  * time rate-of-change k2 evaluated halfway through time step (midpoint) */
78  Eigen::VectorXd v3 = v1;
79  v3.noalias() += m_c1_2_dt * a2;
80  Eigen::VectorXd x3 = x1;
81  x3.noalias() += m_c1_2_dt * v3;
82 
83  m_system->setVelocitiesParticles(v3);
84  m_system->setPositionsParticles(x3);
85 
86  Eigen::VectorXd a3 = Eigen::VectorXd::Zero(3 * m_system->numParticles());
87  accelerationUpdate(a3, device);
88 
89  /* Step 4: k4 = f( y(t_0) + k3 * dt, t_0 + dt )
90  * time rate-of-change k3 evaluated at end of step (endpoint) */
91  Eigen::VectorXd v4 = v1;
92  v4.noalias() += m_dt * a3;
93  Eigen::VectorXd x4 = x1;
94  x4.noalias() += m_dt * v4;
95 
96  m_system->setVelocitiesParticles(v4);
97  m_system->setPositionsParticles(x4);
98 
99  m_system->setT(t1 + m_system->dt());
100 
101  Eigen::VectorXd a4 = Eigen::VectorXd::Zero(3 * m_system->numParticles());
102  accelerationUpdate(a4, device);
103 
104  /* ANCHOR: Calculate kinematics at end of time step */
105  Eigen::VectorXd v_out = a1;
106  v_out.noalias() += 2.0 * a2;
107  v_out.noalias() += 2.0 * a3;
108  v_out.noalias() += a4;
109  v_out *= m_c1_6_dt;
110  v_out.noalias() += v1;
111 
112  Eigen::VectorXd x_out = v1;
113  x_out.noalias() += 2.0 * v2;
114  x_out.noalias() += 2.0 * v3;
115  x_out.noalias() += v4;
116  x_out *= m_c1_6_dt;
117  x_out.noalias() += x1;
118 
119  m_system->setVelocitiesParticles(v_out);
120  m_system->setPositionsParticles(x_out);
121 
122  Eigen::VectorXd a_out = Eigen::VectorXd::Zero(3 * m_system->numParticles());
123  accelerationUpdate(a_out, device);
124  m_system->setAccelerationsParticles(a_out);
125 
126  m_system->setT(t1);
127 }

139 void
140 rungeKutta4::udwadiaKalaba(Eigen::VectorXd& acc)
141 {
142  /* NOTE: Following the formalism developed in Udwadia & Kalaba (1992) Proc. R. Soc. Lond. A
143  * Solve system of the form M_total * acc = Q + Q_con
144  * Q is the forces present in unconstrained system
145  * Q_con is the generalized constraint forces */
146 
147  // calculate Q
148  Eigen::VectorXd Q = Eigen::VectorXd::Zero(3 * m_system->numParticles());
149  if (m_system->fluidDensity() > 0) // hydrodynamic force
150  {
151  Q.noalias() += m_potHydro->fHydroNoInertia();
152  }
153 
154  /* calculate Q_con = K (b - A * M_total^{-1} * Q)
155  * Linear constraint system: A * acc = b
156  * Linear proportionality: K = M_total^{1/2} * (A * M_total^{-1/2})^{+};
157  * + is Moore-Penrose inverse */
158 
159  // calculate M^{1/2} & M^{-1/2}
160  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> eigensolver(m_potHydro->mTotal());
161  if (eigensolver.info() != Eigen::Success)
162  {
163  spdlog::get(m_logName)->error("Computing eigendecomposition of M_total failed at t={0}", m_system->t());
164  throw std::runtime_error("Computing eigendecomposition of M_total failed");
165  }
166  const Eigen::MatrixXd M_total_halfPower = eigensolver.operatorSqrt();
167  const Eigen::MatrixXd M_total_negativeHalfPower = eigensolver.operatorInverseSqrt();
168 
169  // calculate K
170  const Eigen::MatrixXd AM_nHalf = m_system->udwadiaA() * M_total_negativeHalfPower;
171  const Eigen::MatrixXd AM_nHalf_pInv = AM_nHalf.completeOrthogonalDecomposition().pseudoInverse();
172  const Eigen::MatrixXd K = M_total_halfPower * AM_nHalf_pInv;
173 
174  // calculate Q_con
175  const Eigen::MatrixXd M_total_inv = m_potHydro->mTotal().inverse();
176  const Eigen::MatrixXd M_total_invQ = M_total_inv * Q;
177  const Eigen::VectorXd AM_total_invQ = m_system->udwadiaA() * M_total_invQ;
178  Eigen::VectorXd b_tilde = m_system->udwadiaB();
179  b_tilde.noalias() -= AM_total_invQ;
180  const Eigen::VectorXd Q_con = K * b_tilde;
181 
182  // calculate accelerations
183  Eigen::VectorXd Q_total = Q;
184  Q_total.noalias() += Q_con;
185  acc.noalias() = m_potHydro->mTotal().llt().solve(Q_total);
186 }
