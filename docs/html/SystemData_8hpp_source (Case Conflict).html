273  void
274  chiMatrixElement(const double body_id, const double particle_id, Eigen::Matrix<double, 7, 3>& chi_matrix_element)
275  {
276  /* ANCHOR: Compute G matrix element */
277  // body unit quaternion
278  const Eigen::Vector4d theta_body = m_positions_bodies.segment<4>(7 * body_id);
279  // particle unit initial configuration
280  Eigen::Vector4d r_hat_init_particle = Eigen::Vector4d::Zero(4);
281  r_hat_init_particle.segment<3>(1) = m_positions_particles_articulation_init_norm.segment<3>(3 * particle_id);
282 
283  // Q matrices
284  Eigen::Matrix4d q2;
285  Eigen::Matrix4d q3;
286  Eigen::Matrix4d q4;
287 
288  // clang-format off
289  q2 <<
290  theta_body(1), 0, 0, 0,
291  0, theta_body(1), theta_body(2), theta_body(3),
292  0, theta_body(2), -theta_body(1), -theta_body(0),
293  0, theta_body(3), theta_body(0), -theta_body(1);
294 
295  q3 <<
296  theta_body(2), 0, 0, 0,
297  0, -theta_body(2), theta_body(1), theta_body(0),
298  0, theta_body(1), theta_body(2), theta_body(3),
299  0, -theta_body(0), theta_body(3), -theta_body(2);
300 
301  q4 <<
302  theta_body(3), 0, 0, 0,
303  0, -theta_body(3), -theta_body(0), theta_body(1),
304  0, theta_body(0), -theta_body(3), theta_body(2),
305  0, theta_body(1), theta_body(2), theta_body(3);
306  // clang-format on
307 
308  // 2 * || r_particle_id ||, G_matrix prefactor
309  const double prefactor{2 * m_positions_particles_articulation.segment<3>(3 * particle_id).norm()};
310 
311  // G matrix element
312  Eigen::Matrix<double, 4, 3> g_matrix;
313  g_matrix.col(1).noalias() = q2 * r_hat_init_particle;
314  g_matrix.col(2).noalias() = q3 * r_hat_init_particle;
315  g_matrix.col(3).noalias() = q4 * r_hat_init_particle;
316  g_matrix *= prefactor;
317 
318  /* ANCHOR: Compute S matrix element */
319  const bool is_locater{m_particle_group_id(particle_id) == 1}; // determine if particle is locater particle
321  const int s_part{-1 + 2 * is_locater};
322  const Eigen::Matrix3d s_matrix = s_part * m_I3;
323 
324  /* ANCHOR: Compute chi matrix element */
325  chi_matrix_element.block<3, 3>(0, 0).noalias() =
326  s_matrix; // convert body (linear) position derivatives to particle linear coordinate derivatives
327  chi_matrix_element.block<4, 3>(3, 0).noalias() =
328  g_matrix; // convert body (quaternion) position derivatives to particle linear coordinate derivatives
329  }
