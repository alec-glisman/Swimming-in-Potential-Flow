10potentialHydrodynamics::potentialHydrodynamics(std::shared_ptr<systemData> sys)
11{
12 // save classes
13 m_system = sys;
14
15 // Initialize logger
16 m_logFile = m_system->outputDir() + "/logs/" + m_logName + "-log.txt";
17 auto logger = spdlog::basic_logger_mt(m_logName, m_logFile);
18 spdlog::get(m_logName)->info("Initializing potential hydrodynamics");
19
20 // Variables for for-loop
21 m_num_pair_inter = m_system->numParticles() * (m_system->numParticles() - 1) / 2; // Number of interactions to count
22 spdlog::get(m_logName)->info("Setting number of interactions to count: {0}", m_num_pair_inter);
23
24 // tensor variables
25 m_3N = 3 * m_system->numParticles(); // length of tensor quantities
26 spdlog::get(m_logName)->info("Length of tensor quantities: {0}", m_3N);
27
28 // set identity matrices
29 m_I3N = Eigen::MatrixXd::Identity(m_3N, m_3N);
30 m_c1_2_I3N = m_c1_2 * m_I3N;
31
32 // Initialize mass matrices
33 spdlog::get(m_logName)->info("Initializing mass matrices");
34 m_M_intrinsic = (m_system->particleDensity() * m_unitSphereVol) * m_I3N;
35 m_M_added = Eigen::MatrixXd::Zero(m_3N, m_3N);
36
37 m_M_total.noalias() = m_M_added;
38 m_M_total.noalias() += m_M_intrinsic;
39
40 spdlog::get(m_logName)->info("Initializing mass tensors");
41 m_grad_M_added = Eigen::Tensor<double, 3>(m_3N, m_3N, m_3N);
42
43 // Initialize force vectors
44 spdlog::get(m_logName)->info("Initializing hydrodynamic force vectors");
45 m_F_hydro = Eigen::VectorXd::Zero(m_3N);
46 m_F_hydroNoInertia = Eigen::VectorXd::Zero(m_3N);
47 m_t1_Inertia = Eigen::VectorXd::Zero(m_3N);
48
49 spdlog::get(m_logName)->info("Initializing hydrodynamic force tensors");
50 m_t2_VelGrad = Eigen::Tensor<double, 1>(m_3N);
51 m_t2_VelGrad.setZero();
52 m_t3_PosGrad = Eigen::Tensor<double, 1>(m_3N);
53 m_t3_PosGrad.setZero();
54
55 // Assign particle pair information
56 spdlog::get(m_logName)->info("Initializing particle pair information vectors");
57 m_alphaVec = Eigen::VectorXd::Zero(m_num_pair_inter);
58 m_betaVec = Eigen::VectorXd::Zero(m_num_pair_inter);
59 m_r_mag_ab = Eigen::VectorXd::Zero(m_num_pair_inter);
60 m_r_ab = Eigen::MatrixXd::Zero(3, m_num_pair_inter);
61
62 /* Fill the particle index vectors
63 * Calculate ahead of time to save time during runtime
64 */
65 spdlog::get(m_logName)->info("Filling particle pair information tensors");
66 for (int i = 0; i < m_num_pair_inter; i++)
67 {
68 /* alpha and beta convention to convert from linear coordinate to ordered pair
69 * @Source:
70 * https://stackoverflow.com/questions/33810187/openmp-and-c-private-variables/33836073#33836073
71 */
72 int alpha = i / m_system->numParticles();
73 int beta = i % m_system->numParticles();
74
75 if (beta <= alpha)
76 {
77 alpha = m_system->numParticles() - alpha - 2;
78 beta = m_system->numParticles() - beta - 1;
79 }
80
81 assert(alpha >= 0 && "Calculated alpha must be non-negative");
82 assert(beta >= 0 && "Calculated beta must be non-negative");
83
84 m_alphaVec[i] = alpha;
85 m_betaVec[i] = beta;
86 }
87
88 // Compute all relevant quantities
89 spdlog::get(m_logName)->info("Calling update()");
90 update();
91
92 spdlog::get(m_logName)->info("Constructor complete");
93 spdlog::get(m_logName)->flush();
94}

114void
115potentialHydrodynamics::calcParticleDistances()
116{
117 /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
118 * particles \alpha and \beta) */
119 for (int i = 0; i < m_num_pair_inter; i++)
120 {
121 m_r_ab.col(i).noalias() = m_system->positionsParticles()(Eigen::seqN(3 * m_alphaVec[i], 3));
122 m_r_ab.col(i).noalias() -= m_system->positionsParticles()(Eigen::seqN(3 * m_betaVec[i], 3));
123
124 m_r_mag_ab[i] = m_r_ab.col(i).norm(); //[1]; |r| between 2 particles
125
126#if !defined(NDEBUG) && !defined(NO_HYDRO_CHECKS)
127 spdlog::get(m_logName)->critical("Checking distance between pairs {0} & {1}", m_alphaVec[i], m_betaVec[i]);
128 spdlog::get(m_logName)->critical("Interparticle distance is", m_r_mag_ab[i]);
129 spdlog::get(m_logName)->flush();
130#endif
131 }

134void
135potentialHydrodynamics::calcAddedMass()
136{
137 // set matrices to zero
138 m_M_added.setZero(m_3N, m_3N);
139
140 /* Fill off-diagonal elements (without units ) */
141 /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
142 * particles \alpha and \beta) */
143 for (int k = 0; k < m_num_pair_inter; k++)
144 {
145 // Convert (\alpha, \beta) --> (i, j) by factor of 3
146 int i_part = 3 * m_alphaVec[k];
147 int j_part = 3 * m_betaVec[k];
148
149 // Full distance between particles \alpha and \beta
150 Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
151 double r_mag_ij = m_r_mag_ab[k]; //[1]; |r| between 2 particles
152
153 // M^{(1)} Matrix Element Constants:
154 double M1_c1 = -m_c3_2 / std::pow(r_mag_ij, 5); // [1]
155 double M1_c2 = m_c1_2 / std::pow(r_mag_ij, 3); // [1]
156
157 // Full matrix elements for M^{(1)}_{ij} (NOTE: missing factor of 1/2)
158 Eigen::Matrix3d Mij = r_ij * r_ij.transpose(); //[1]; Outer product of \bm{r} \bm{r}
159 Mij *= M1_c1;
160 Mij.noalias() += M1_c2 * m_system->i3();
161
162 // Output added mass element (symmetry of mass matrix)
163 m_M_added.block<3, 3>(i_part, j_part).noalias() = Mij;
164 m_M_added.block<3, 3>(j_part, i_part).noalias() = Mij;
165 }
166
167 /* Construct full added mass matrix
168 * M = 1/2 I + M^{(1)}
169 */
170 m_M_added.noalias() += m_c1_2_I3N; // Add diagonal elements
171 m_M_added *= (m_system->fluidDensity() * m_unitSphereVol); // mass units
172}

174void
175potentialHydrodynamics::calcAddedMassGrad()
176{
177 // eigen tensor contraction variables
178 Eigen::array<Eigen::IndexPair<long>, 0> empty_index_list = {};
179 Eigen::array<int, 3> shuffle_one_right({2, 0, 1});
180 Eigen::array<int, 3> shuffle_one_left({1, 2, 1});
181
182 // set matrices to zero
183 m_grad_M_added.setZero();
184
185 /* NOTE: Fill Mass matrix elements one (3 x 3 x 3) block at a time (matrix elements between
186 * particles \alpha and \beta) */
187
188 for (int k = 0; k < m_num_pair_inter; k++)
189 {
190 // Convert (\alpha, \beta) --> (i, j) by factor of 3
191 int i_part = 3 * m_alphaVec[k];
192 int j_part = 3 * m_betaVec[k];
193
194 // Full distance between particles \alpha and \beta
195 Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
196 double r_mag_ij = m_r_mag_ab[k]; //[1]; |r| between 2 particles
197 Eigen::Matrix3d r_dyad_r = r_ij * r_ij.transpose(); // [1]; Outer product of \bm{r} \bm{r}
198 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_rr = TensorCast(r_dyad_r);
199
200 // Constants to use in Calculation
201 double gradM1_c1 =
202 -(m_system->fluidDensity() * m_unitSphereVol) * m_c3_2 * std::pow(r_mag_ij, -5); // mass units
203 double gradM1_c2 =
204 (m_system->fluidDensity() * m_unitSphereVol) * m_c15_2 * std::pow(r_mag_ij, -7); // mass units
205
206 // outer products (I_{i j} r_{k}) and permutations
207 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ij_r_k =
208 gradM1_c1 * m_system->tensI3().contract(TensorCast(r_ij), empty_index_list);
209 // shuffle all dimensions to the right by 1: (i, j, k) --> (k, i, j), (2, 0, 1)
210 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_jk_r_i = delta_ij_r_k.shuffle(shuffle_one_right);
211 // shuffle all dimensions to the left by 1: (i, j, k) --> (k, i, j), (1, 2, 0)
212 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ki_r_j = delta_ij_r_k.shuffle(shuffle_one_left);
213
214 // full matrix element for M_{i j, i}
215 Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> Mij_i = delta_ij_r_k + delta_jk_r_i;
216 Mij_i += delta_ki_r_j;
217 Mij_i += gradM1_c2 * tens_rr.contract(TensorCast(r_ij), empty_index_list);
218
219 // indices to start at
220 Eigen::array<Eigen::Index, 3> offsets_ij_i = {i_part, j_part, i_part};
221 Eigen::array<Eigen::Index, 3> offsets_ij_j = {i_part, j_part, j_part};
222 Eigen::array<Eigen::Index, 3> offsets_ji_i = {j_part, i_part, i_part};
223 Eigen::array<Eigen::Index, 3> offsets_ji_j = {j_part, i_part, j_part};
224 // length of data to access
225 Eigen::array<Eigen::Index, 3> extents = {3, 3, 3};
226
227 // M_{ij, i}: Matrix Element (Anti-Symmetric upon exchange of derivative, Symmetric upon
228 // exchange of first two indices)
229 m_grad_M_added.slice(offsets_ij_i, extents) = Mij_i;
230
231 // M_{ij, j} = - M_{ij, i}
232 m_grad_M_added.slice(offsets_ij_j, extents) = -Mij_i;
233
234 // M_{ji, j} = - M_{ij, i}
235 m_grad_M_added.slice(offsets_ji_j, extents) = -Mij_i;
236
237 // M_{ji, i} = M_{ij, i}
238 m_grad_M_added.slice(offsets_ji_i, extents) = Mij_i;
239 }

250void
251potentialHydrodynamics::calcHydroForces()
252{
253 // calculate requisite configurational tensors
254 Eigen::MatrixXd U_dyad_U = m_system->velocitiesParticles() * m_system->velocitiesParticles().transpose();
255 Eigen::Tensor<double, 2> tens_UU = TensorCast(U_dyad_U);
256
257 // Term 1: - M_kj * U_dot_j
258 m_t1_Inertia.noalias() = -m_M_total * m_system->accelerationsParticles(); // dim = (3N) x 1
259
260 // Term 2: - d(M_kj)/d(R_l) * U_l * U_j;
261 Eigen::array<Eigen::IndexPair<int>, 2> contract_kjl_jl = {Eigen::IndexPair<int>(1, 0), Eigen::IndexPair<int>(2, 1)};
262 m_t2_VelGrad = -m_grad_M_added.contract(tens_UU, contract_kjl_jl);
263
264 // Term 3: (1/2) U_i * d(M_ij)/d(R_k) * U_j
265 Eigen::array<Eigen::IndexPair<int>, 2> contract_ijk_ij = {Eigen::IndexPair<int>(0, 0), Eigen::IndexPair<int>(1, 1)};
266 m_t3_PosGrad = 0.50 * m_grad_M_added.contract(tens_UU, contract_ijk_ij);
267
268 // Compute non-inertial part of force
269 m_F_hydroNoInertia.noalias() = MatrixCast(m_t2_VelGrad, m_3N, 1);
270 m_F_hydroNoInertia.noalias() += MatrixCast(m_t3_PosGrad, m_3N, 1);
271
272 // Compute complete potential pressure force
273 m_F_hydro.noalias() = m_t1_Inertia; // Full hydrodynamic force, dim = (3N) x 1
274 m_F_hydro.noalias() += m_F_hydroNoInertia;
