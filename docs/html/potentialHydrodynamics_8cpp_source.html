142 potentialHydrodynamics::calcParticleDistances()
143 {
144  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
145  * particles \alpha and \beta) */
146  for (int i = 0; i < m_num_pair_inter; i++)
147  {
148  m_r_ab.col(i).noalias() = m_system->positionsParticles()(Eigen::seqN(3 * m_alphaVec[i], 3));
149  m_r_ab.col(i).noalias() -= m_system->positionsParticles()(Eigen::seqN(3 * m_betaVec[i], 3));
150

161 void
162 potentialHydrodynamics::calcAddedMass()
163 {
164  // set matrices to zero
165  m_M_added.setZero(m_3N, m_3N);
166 
167  /* Fill off-diagonal elements (without units ) */
168  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
169  * particles \alpha and \beta) */
170  for (int k = 0; k < m_num_pair_inter; k++)
171  {
172  // Convert (\alpha, \beta) --> (i, j) by factor of 3
173  int i_part = 3 * m_alphaVec[k];
174  int j_part = 3 * m_betaVec[k];
175 
176  // Full distance between particles \alpha and \beta
177  Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
178  double r_mag_ij = m_r_mag_ab[k]; //[1]; |r| between 2 particles
179 
180  // M^{(1)} Matrix Element Constants:
181  double M1_c1 = -m_c3_2 / std::pow(r_mag_ij, 5); // [1]
182  double M1_c2 = m_c1_2 / std::pow(r_mag_ij, 3); // [1]
183 
184  // Full matrix elements for M^{(1)}_{ij} (NOTE: missing factor of 1/2)
185  Eigen::Matrix3d Mij = r_ij * r_ij.transpose(); //[1]; Outer product of \bm{r} \bm{r}
186  Mij *= M1_c1;
187  Mij.noalias() += M1_c2 * m_system->i3();
188 
189  // Output added mass element (symmetry of mass matrix)
190  m_M_added.block<3, 3>(i_part, j_part).noalias() = Mij;
191  m_M_added.block<3, 3>(j_part, i_part).noalias() = Mij;
192  }
193 
194  /* Construct full added mass matrix
195  * M = 1/2 I + M^{(1)}
196  */
197  m_M_added.noalias() += m_c1_2_I3N; // Add diagonal elements
198  m_M_added *= (m_system->fluidDensity() * m_unitSphereVol); // mass units
199 }

201 void
202 potentialHydrodynamics::calcAddedMassGrad(Eigen::ThreadPoolDevice& device)
203 {
204  // eigen tensor contraction variables
205  Eigen::array<Eigen::IndexPair<int>, 1> contract_ik_kj = {Eigen::IndexPair<int>(1, 0)}; // = A B
206  Eigen::array<Eigen::IndexPair<long>, 0> empty_index_list = {};
207  Eigen::array<int, 3> shuffle_one_right({2, 0, 1});
208  Eigen::array<int, 3> shuffle_one_left({1, 2, 1});
209 
210  // set matrices to zero
211  m_grad_M_added.setZero();
212 
213  /* NOTE: Fill Mass matrix elements one (3 x 3 x 3) block at a time (matrix elements between
214  * particles \alpha and \beta) */
215 
216  for (int k = 0; k < m_num_pair_inter; k++)
217  {
218  // Convert (\alpha, \beta) --> (i, j) by factor of 3
219  int i_part = 3 * m_alphaVec[k];
220  int j_part = 3 * m_betaVec[k];
221 
222  // Full distance between particles \alpha and \beta
223  Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
224  double r_mag_ij = m_r_mag_ab[k]; //[1]; |r| between 2 particles
225  Eigen::Matrix3d r_dyad_r = r_ij * r_ij.transpose(); // [1]; Outer product of \bm{r} \bm{r}
226  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_rr = TensorCast(r_dyad_r);
227 
228  // Constants to use in Calculation
229  double gradM1_c1 =
230  -(m_system->fluidDensity() * m_unitSphereVol) * m_c3_2 * std::pow(r_mag_ij, -5); // mass units
231  double gradM1_c2 =
232  (m_system->fluidDensity() * m_unitSphereVol) * m_c15_2 * std::pow(r_mag_ij, -7); // mass units
233 
234  // outer products (I_{i j} r_{k}) and permutations
235  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ij_r_k =
236  gradM1_c1 * m_system->tensI3().contract(TensorCast(r_ij), empty_index_list);
237  // shuffle all dimensions to the right by 1: (i, j, k) --> (k, i, j), (2, 0, 1)
238  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_jk_r_i = delta_ij_r_k.shuffle(shuffle_one_right);
239  // shuffle all dimensions to the left by 1: (i, j, k) --> (k, i, j), (1, 2, 0)
240  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ki_r_j = delta_ij_r_k.shuffle(shuffle_one_left);
241 
242  // full matrix element for M_{i j, i}
243  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> Mij_i = delta_ij_r_k + delta_jk_r_i;
244  Mij_i += delta_ki_r_j;
245  Mij_i += gradM1_c2 * tens_rr.contract(TensorCast(r_ij), empty_index_list);
246 
247  // indices to start at
248  Eigen::array<Eigen::Index, 3> offsets_ij_i = {i_part, j_part, i_part};
249  Eigen::array<Eigen::Index, 3> offsets_ij_j = {i_part, j_part, j_part};
250  Eigen::array<Eigen::Index, 3> offsets_ji_i = {j_part, i_part, i_part};
251  Eigen::array<Eigen::Index, 3> offsets_ji_j = {j_part, i_part, j_part};
252  // length of data to access
253  Eigen::array<Eigen::Index, 3> extents = {3, 3, 3};
254 
255  // M_{ij, i}: Matrix Element (Anti-Symmetric upon exchange of derivative, Symmetric upon
256  // exchange of first two indices)
257  m_grad_M_added.slice(offsets_ij_i, extents) = Mij_i;
258 
259  // M_{ij, j} = - M_{ij, i}
260  m_grad_M_added.slice(offsets_ij_j, extents) = -Mij_i;
261 
262  // M_{ji, j} = - M_{ij, i}
263  m_grad_M_added.slice(offsets_ji_j, extents) = -Mij_i;
264 
265  // M_{ji, i} = M_{ij, i}
266  m_grad_M_added.slice(offsets_ji_i, extents) = Mij_i;
267  }
268 
269  m_grad_M_added_body_coords.device(device) =
270  m_system->tensConvBody2PartDof().contract(m_grad_M_added, contract_ik_kj);
271 }

299 void
300 potentialHydrodynamics::calcHydroForces(Eigen::ThreadPoolDevice& device)
301 {
302  // tensor contraction indices
303  Eigen::array<Eigen::IndexPair<int>, 2> contract_ijk_jk = {Eigen::IndexPair<int>(1, 0), Eigen::IndexPair<int>(2, 1)};
304  Eigen::array<Eigen::IndexPair<int>, 2> contract_ijk_ij = {Eigen::IndexPair<int>(0, 0), Eigen::IndexPair<int>(1, 1)};
305 
306  // calculate requisite configurational tensors
307  Eigen::MatrixXd U_dyad_U = m_system->velocitiesParticles() * m_system->velocitiesParticles().transpose();
308  Eigen::Tensor<double, 2> tens_UU = TensorCast(U_dyad_U);
309 
310  // Term 1: - M_kj * U_dot_j
311  m_t1_Inertia.noalias() = -m_M_total * m_system->accelerationsParticles(); // dim = (3N) x 1
312 
313  // Term 2: - d(M_kj)/d(R_l) * U_l * U_j;
314  m_t2_VelGrad.device(device) = -m_grad_M_added.contract(tens_UU, contract_ijk_jk);
315 
316  // Term 3: (1/2) U_i * d(M_ij)/d(R_k) * U_j
317  m_t3_PosGrad.device(device) = 0.50 * m_grad_M_added.contract(tens_UU, contract_ijk_ij);
318 
319  // Compute non-inertial part of force
320  m_F_hydroNoInertia.noalias() = MatrixCast(m_t2_VelGrad, m_3N, 1);
321  m_F_hydroNoInertia.noalias() += MatrixCast(m_t3_PosGrad, m_3N, 1);
322 
323  // Compute complete potential pressure force
324  m_F_hydro.noalias() = m_t1_Inertia; // Full hydrodynamic force, dim = (3N) x 1
325  m_F_hydro.noalias() += m_F_hydroNoInertia;
326 }
