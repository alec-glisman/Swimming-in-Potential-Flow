117 potentialHydrodynamics::calcParticleDistances()
118 {
119  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
120  * particles \alpha and \beta) */
121  for (int i = 0; i < m_num_inter; i++)
122  {
123  m_r_ab.col(i).noalias() = m_system->positionsParticles()(Eigen::seqN(3 * m_alphaVec[i], 3));
124  m_r_ab.col(i).noalias() -= m_system->positionsParticles()(Eigen::seqN(3 * m_betaVec[i], 3));
125

137 void
138 potentialHydrodynamics::calcAddedMass()
139 {
140  // set matrices to zero
141  m_M_added.setZero(m_len_tensor, m_len_tensor);
142 
143  /* Fill off-diagonal elements (without units ) */
144  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
145  * particles \alpha and \beta) */
146  for (int k = 0; k < m_num_inter; k++)
147  {
149  int i_part = 3 * m_alphaVec[k];
150  int j_part = 3 * m_betaVec[k];
151 
153  Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
154  double r_mag_ij = m_r_mag_ab[k];
155 
157  double M1_c1 = -m_c3_2 / std::pow(r_mag_ij, 5); // [1]
158  double M1_c2 = m_c1_2 / std::pow(r_mag_ij, 3); // [1]
159 
161  Eigen::Matrix3d Mij = r_ij * r_ij.transpose();
162  Mij *= M1_c1;
163  Mij.noalias() += M1_c2 * m_I3;
164 
166  m_M_added.block<3, 3>(i_part, j_part).noalias() = Mij;
167  m_M_added.block<3, 3>(j_part, i_part).noalias() = Mij;
168  }
169 
170  /* Construct full added mass matrix
171  * M = 1/2 I + M^{(1)}
172  */
173  m_M_added.noalias() += m_c1_2_I3N; // Add diagonal elements
174  m_M_added *= (m_system->fluidDensity() * m_unitSphereVol); // mass units
175 }

178 potentialHydrodynamics::calcAddedMassGrad()
179 {
180  // set matrices to zero
181  m_grad_M_added.setZero(m_len_tensor, m_len_tensor * m_len_tensor);
182 
183  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
184  * particles \alpha and \beta) */
185  for (int k = 0; k < m_num_inter; k++)
186  {
188  int i_part = 3 * m_alphaVec[k];
189  int j_part = 3 * m_betaVec[k];
190 
192  Eigen::Vector3d r_ij = m_r_ab.col(k); // [1]
193  double r_mag_ij = m_r_mag_ab[k];
194 
196  Eigen::Matrix3d delta_Ri_x =
197  Eigen::Matrix3d::Zero();
198  Eigen::Matrix3d delta_Ri_y =
199  Eigen::Matrix3d::Zero();
200  Eigen::Matrix3d delta_Ri_z =
201  Eigen::Matrix3d::Zero();
202 
203  /* (\delta_{j x} r_i) Adds all components of \bm{r} as a column vector to the first column
204  * (j=x) The indexing comes from a M_{i j, k} index scheme, where k=x in this case */
205  delta_Ri_x.col(0).noalias() += r_ij;
206  delta_Ri_x.row(0).noalias() += r_ij.transpose();
207  delta_Ri_y.col(1).noalias() += r_ij;
208  delta_Ri_y.row(1).noalias() += r_ij.transpose();
209  delta_Ri_z.col(2).noalias() += r_ij;
210  delta_Ri_z.row(2).noalias() += r_ij.transpose();
211 
213  double gradM1_c1 = -m_c3_2 / std::pow(r_mag_ij, 5); // [1]
214  double gradM1_c2 = m_c15_2 / std::pow(r_mag_ij, 7); // [1]
215 
217  Eigen::Matrix3d r_dyad_r = r_ij * r_ij.transpose(); // [1]; Outer product of \bm{r} \bm{r}
218 
219  // Mij_ix
220  Eigen::Matrix3d Mij_ix = delta_Ri_x;
221  Mij_ix.noalias() += r_ij[0] * m_I3;
222  Mij_ix *= gradM1_c1; // sub-term 1 complete
223  Mij_ix.noalias() += (gradM1_c2 * r_ij[0]) * r_dyad_r; // sub-term 2 complete
224 
225  // Mij_iy
226  Eigen::Matrix3d Mij_iy = delta_Ri_y;
227  Mij_iy.noalias() += r_ij[1] * m_I3;
228  Mij_iy *= gradM1_c1; // sub-term 1 complete
229  Mij_iy.noalias() += (gradM1_c2 * r_ij[1]) * r_dyad_r; // sub-term 2 complete
230 
231  // Mij_iz
232  Eigen::Matrix3d Mij_iz = delta_Ri_z;
233  Mij_iz.noalias() += r_ij[2] * m_I3;
234  Mij_iz *= gradM1_c1; // sub-term 1 complete
235  Mij_iz.noalias() += (gradM1_c2 * r_ij[2]) * r_dyad_r; // sub-term 2 complete
236 
238  int row_Ri = i_part;
239  int row_Rj = j_part;
240 
242  int col_j_dRi_x = m_system->convert3dIdxTo2d(j_part, i_part, 0, m_len_tensor);
243  int col_j_dRi_y = m_system->convert3dIdxTo2d(j_part, i_part, 1, m_len_tensor);
244  int col_j_dRi_z = m_system->convert3dIdxTo2d(j_part, i_part, 2, m_len_tensor);
245 
247  int col_j_dRj_x = m_system->convert3dIdxTo2d(j_part, j_part, 0, m_len_tensor);
248  int col_j_dRj_y = m_system->convert3dIdxTo2d(j_part, j_part, 1, m_len_tensor);
249  int col_j_dRj_z = m_system->convert3dIdxTo2d(j_part, j_part, 2, m_len_tensor);
250 
252  int col_i_dRj_x = m_system->convert3dIdxTo2d(i_part, j_part, 0, m_len_tensor);
253  int col_i_dRj_y = m_system->convert3dIdxTo2d(i_part, j_part, 1, m_len_tensor);
254  int col_i_dRj_z = m_system->convert3dIdxTo2d(i_part, j_part, 2, m_len_tensor);
255 
257  int col_i_dRi_x = m_system->convert3dIdxTo2d(i_part, i_part, 0, m_len_tensor);
258  int col_i_dRi_y = m_system->convert3dIdxTo2d(i_part, i_part, 1, m_len_tensor);
259  int col_i_dRi_z = m_system->convert3dIdxTo2d(i_part, i_part, 2, m_len_tensor);
260 
263  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRi_x).noalias() = Mij_ix;
264  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRi_y).noalias() = Mij_iy;
265  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRi_z).noalias() = Mij_iz;
266 
268  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRj_x).noalias() = -Mij_ix;
269  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRj_y).noalias() = -Mij_iy;
270  m_grad_M_added.block<3, 3>(row_Ri, col_j_dRj_z).noalias() = -Mij_iz;
271 
273  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRj_x).noalias() = -Mij_ix;
274  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRj_y).noalias() = -Mij_iy;
275  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRj_z).noalias() = -Mij_iz;
276 
278  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRi_x).noalias() = Mij_ix;
279  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRi_y).noalias() = Mij_iy;
280  m_grad_M_added.block<3, 3>(row_Rj, col_i_dRi_z).noalias() = Mij_iz;
281  }
282 
283  /* Construct full added mass matrix
284  * \nabla M = \nabla M^{(1)}
285  */

297 void
298 potentialHydrodynamics::calcHydroForces()
299 {
300  // calculate requisite configurational tensors
301  Eigen::MatrixXd U_dyad_U =
302  m_system->velocitiesParticles() * m_system->velocitiesParticles().transpose();
303 
305  m_t1_Inertia.noalias() = -m_M_total * m_system->accelerationsParticles();
306 
308  Eigen::MatrixXd hat_dMkj = Eigen::MatrixXd::Zero(m_len_tensor, m_len_tensor);
309  // Reduce along derivative vars (index: l) via matrix-vector product for each derivative
310  // variable
311  for (int l = 0; l < m_len_tensor; l += 1)
312  {
313  hat_dMkj.noalias() += m_system->velocitiesParticles()(l) *
314  m_grad_M_added.block(0, l * m_len_tensor, m_len_tensor, m_len_tensor);
315  }
316  // Reduce along gradient (index: l) via matrix-vector product
317  m_t2_VelGrad.noalias() = -hat_dMkj * m_system->velocitiesParticles();
318 
320  m_t3_PosGrad.noalias() = Eigen::VectorXd::Zero(m_len_tensor);
321 
322  for (int k = 0; k < m_len_tensor; k += 3)
323  {
324  // Unroll loop slightly (do 3 entries at a time)
325  m_t3_PosGrad(k) = m_grad_M_added.block(0, k * m_len_tensor, m_len_tensor, m_len_tensor)
326  .cwiseProduct(U_dyad_U)
327  .sum();
328  m_t3_PosGrad(k + 1) =
329  m_grad_M_added.block(0, (k + 1) * m_len_tensor, m_len_tensor, m_len_tensor)
330  .cwiseProduct(U_dyad_U)
331  .sum();
332  m_t3_PosGrad(k + 2) =
333  m_grad_M_added.block(0, (k + 2) * m_len_tensor, m_len_tensor, m_len_tensor)
334  .cwiseProduct(U_dyad_U)
335  .sum();
336  }
337  /* Coefficient-wise operations for the sub-terms */
338  m_t3_PosGrad *= m_c1_2;
339 
340  /* Compute non-inertial part of force */
341  m_F_hydroNoInertia.noalias() = m_t2_VelGrad;
342  m_F_hydroNoInertia.noalias() += m_t3_PosGrad;
343 
344  /* Compute complete potential pressure force */
345  m_F_hydro.noalias() = m_t1_Inertia; // Full hydrodynamic force, dim = (3N) x 1
346  m_F_hydro.noalias() += m_F_hydroNoInertia;
347 }
