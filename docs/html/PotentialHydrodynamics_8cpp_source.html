168 PotentialHydrodynamics::calcParticleDistances()
169 {
170  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
171  * particles \alpha and \beta) */
172  for (int i = 0; i < m_num_pair_inter; i++)
173  {
174  m_r_ab.col(i).noalias() = m_system->positionsParticles().segment<3>(3 * m_alphaVec(i));
175  m_r_ab.col(i).noalias() -= m_system->positionsParticles().segment<3>(3 * m_betaVec(i));
176 
177  m_r_mag_ab(i) = m_r_ab.col(i).norm(); //(1); |r| between 2 particles

179 #if !defined(NDEBUG)

187 void
188 PotentialHydrodynamics::calcAddedMass()
189 {
190  // set matrices to zero
191  m_M_added.setZero();
192 
193  /* Fill off-diagonal elements (without units ) */
194  /* NOTE: Fill Mass matrix elements one (3 x 3) block at a time (matrix elements between
195  * particles \alpha and \beta) */
196  for (int k = 0; k < m_num_pair_inter; k++)
197  {
198  // Convert (\alpha, \beta) --> (i, j) by factor of 3
199  const int i_part{7 * m_alphaVec(k)};
200  const int j_part{7 * m_betaVec(k)};
201 
202  // Full distance between particles \alpha and \beta
203  const Eigen::Vector3d r_ij = m_r_ab.col(k); // (1)
204  const double r_mag_ij{m_r_mag_ab(k)}; //(1); |r| between 2 particles
205 
206  // M^{(1)} Matrix Element Constants:
207  const double M1_c1{-m_c3_2 / std::pow(r_mag_ij, 5)}; // (1)
208  const double M1_c2{m_c1_2 / std::pow(r_mag_ij, 3)}; // (1)
209 
210  // Full matrix elements for M^{(1)}_{ij} (NOTE: missing factor of 1/2)
211  Eigen::Matrix3d Mij = r_ij * r_ij.transpose(); //(1); Outer product of \bm{r} \bm{r}
212  Mij *= M1_c1;
213  Mij.noalias() += M1_c2 * m_system->i3();
214 
215  // Output added mass element (symmetry of mass matrix)
216  m_M_added.block<3, 3>(i_part, j_part).noalias() = Mij;
217  m_M_added.block<3, 3>(j_part, i_part).noalias() = Mij;
218  }
219 
220  /* Construct full added mass matrix
221  * M = 1/2 I + M^{(1)}
222  */
223  m_M_added.noalias() += m_c1_2_I7N_linear; // Add diagonal elements
224  m_M_added *= (m_system->fluidDensity() * m_unit_sphere_volume); // mass units
225 }

227 void
228 PotentialHydrodynamics::calcAddedMassGrad(Eigen::ThreadPoolDevice& device)
229 {
230  // `Eigen::Tensor` contraction indices
231  const Eigen::array<Eigen::IndexPair<int>, 1> contract_ijl_kl = {Eigen::IndexPair<int>(2, 1)};
232  const Eigen::array<Eigen::IndexPair<long>, 0> outer_product = {};
233 
234  // `Eigen::Tensor` permutation indices
235  const Eigen::array<int, 3> permute_ijk_kij({2, 0, 1}); // {i, j, k} --> {k, i, j}
236  const Eigen::array<int, 3> permute_ijk_jki({1, 2, 1}); // {i, j, k} --> {j, k, i}
237 
238  // set matrices to zero
239  m_grad_M_added.setZero();
240 
241  /* NOTE: Fill Mass matrix elements one (3 x 3 x 3) block at a time (matrix elements between
242  * particles \alpha and \beta) */
243 
244  for (int k = 0; k < m_num_pair_inter; k++)
245  {
246  // Convert (\alpha, \beta) --> (i, j) by factor of 3
247  int i_part{7 * m_alphaVec(k)};
248  int j_part{7 * m_betaVec(k)};
249 
250  // Full distance between particles \alpha and \beta
251  const Eigen::Vector3d r_ij = m_r_ab.col(k); // (1)
252  const Eigen::TensorFixedSize<double, Eigen::Sizes<3>> tens_r = TensorCast(r_ij, 3);
253 
254  const double r_mag_ij{m_r_mag_ab(k)}; //(1); |r| between 2 particles
255  const Eigen::Matrix3d r_dyad_r = r_ij * r_ij.transpose(); // (1); Outer product of \bm{r} \bm{r}
256  const Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3>> tens_rr = TensorCast(r_dyad_r, 3, 3);
257 
258  // Constants to use in Calculation
259  double gradM1_c1{-(m_system->fluidDensity() * m_unit_sphere_volume) * m_c3_2 *
260  std::pow(r_mag_ij, -5)}; // mass units
261  double gradM1_c2{(m_system->fluidDensity() * m_unit_sphere_volume) * m_c15_2 *
262  std::pow(r_mag_ij, -7)}; // mass units
263 
264  // outer products (I_{i j} r_{k}) and permutations
265  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ij_r_k;
266  delta_ij_r_k.device(device) = gradM1_c1 * m_system->tensI3().contract(tens_r, outer_product);
267  // shuffle all dimensions to the right by 1: (i, j, k) --> (k, i, j), (2, 0, 1)
268  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_jk_r_i;
269  delta_jk_r_i.device(device) = delta_ij_r_k.shuffle(permute_ijk_kij);
270  // shuffle all dimensions to the left by 1: (i, j, k) --> (k, i, j), (1, 2, 0)
271  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> delta_ki_r_j;
272  delta_ki_r_j.device(device) = delta_ij_r_k.shuffle(permute_ijk_jki);
273 
274  // full matrix element for M_{i j, i}
275  Eigen::TensorFixedSize<double, Eigen::Sizes<3, 3, 3>> Mij_i;
276  Mij_i.device(device) = delta_ij_r_k + delta_jk_r_i;
277  Mij_i.device(device) += delta_ki_r_j;
278  Mij_i.device(device) += gradM1_c2 * tens_rr.contract(tens_r, outer_product);
279 
280  // indices to start at
281  const Eigen::array<Eigen::Index, 3> offsets_ij_i = {i_part, j_part, i_part};
282  const Eigen::array<Eigen::Index, 3> offsets_ij_j = {i_part, j_part, j_part};
283  const Eigen::array<Eigen::Index, 3> offsets_ji_i = {j_part, i_part, i_part};
284  const Eigen::array<Eigen::Index, 3> offsets_ji_j = {j_part, i_part, j_part};
285  // length of data to access
286  const Eigen::array<Eigen::Index, 3> extents = {3, 3, 3};
287 
288  // M_{ij, i}: Matrix Element (Anti-Symmetric upon exchange of derivative, Symmetric upon
289  // exchange of first two indices)
290  m_grad_M_added.slice(offsets_ij_i, extents).device(device) = Mij_i;
291 
292  // M_{ij, j} = - M_{ij, i}
293  m_grad_M_added.slice(offsets_ij_j, extents).device(device) = -Mij_i;
294 
295  // M_{ji, j} = - M_{ij, i}
296  m_grad_M_added.slice(offsets_ji_j, extents).device(device) = -Mij_i;
297 
298  // M_{ji, i} = M_{ij, i}
299  m_grad_M_added.slice(offsets_ji_i, extents).device(device) = Mij_i;
300  }
301 
302  m_grad_M_added_body_coords.device(device) = m_grad_M_added.contract(m_system->tensChi(), contract_ijl_kl);
303 }

316 PotentialHydrodynamics::calcBodyTensors(Eigen::ThreadPoolDevice& device)
317 {
318  /* ANCHOR: Indices */
319  // `Eigen::Tensor` contraction indices
320  const Eigen::array<Eigen::IndexPair<int>, 1> contract_il_lj = {Eigen::IndexPair<int>(1, 0)}; // = A B
321  const Eigen::array<Eigen::IndexPair<int>, 1> contract_il_jl = {Eigen::IndexPair<int>(1, 1)}; // = A B^T
322 
323  // `Eigen::Tensor` permutation indices
324  const Eigen::array<int, 3> permute_ikj_ijk({0, 2, 1}); // (i, j, k) --> (i, j, k)
325 
326  /* ANCHOR: Compute linear combinations of total mass matrix and zeta */
327  m_M2.device(device) = m_system->tensZeta().contract(m_tens_M_total, contract_il_lj);
328 
329  m_M3.device(device) = m_M2.contract(m_system->tensZeta(), contract_il_jl);
330  m_mat_M3 = MatrixCast(m_M3, m_7M, m_7M, device);
331 
332  /* ANCHOR: Compute linear combinations of GRADIENTS of total mass matrix and zeta */
333  // N^{(1)}
334  m_N1.device(device) = m_grad_M_added_body_coords;
335 
336  // N^{(2)}
337  N2_term1_preshuffle.device(device) = m_system->tensGradZeta().contract(m_tens_M_total, contract_il_lj);
338 
339  m_N2.device(device) = N2_term1_preshuffle.shuffle(permute_ikj_ijk);
340  m_N2.device(device) += m_system->tensZeta().contract(m_N1, contract_il_lj);
341 
342  // N^{(3)}
343  N3_term1_preshuffle.device(device) = N2_term1.contract(m_system->tensZeta(), contract_il_jl);
344  N3_term1_preshuffle.device(device) += N2_term2.contract(m_system->tensZeta(), contract_il_jl);
345 
346  m_N3.device(device) = N3_term1_preshuffle.shuffle(permute_ikj_ijk);
347  m_N3.device(device) += m_M2.contract(m_system->tensGradZeta(), contract_il_jl);
348 }

350 void
351 PotentialHydrodynamics::calcHydroForces(Eigen::ThreadPoolDevice& device)
352 {
353  // `Eigen::Tensor` contraction indices
354  // 0th order contraction
355  const Eigen::array<Eigen::IndexPair<int>, 0> outer_product = {};
356  // 1st order contraction
357  const Eigen::array<Eigen::IndexPair<int>, 1> contract_ij_j = {Eigen::IndexPair<int>(1, 0)};
358  // 2nd order contraction
359  const Eigen::array<Eigen::IndexPair<int>, 2> contract_jki_jk = {Eigen::IndexPair<int>(0, 0),
360  Eigen::IndexPair<int>(1, 1)};
361  const Eigen::array<Eigen::IndexPair<int>, 2> contract_ijk_jk = {Eigen::IndexPair<int>(1, 0),
362  Eigen::IndexPair<int>(2, 1)};
363 
364  // `Eigen::Tensor` permutation indices
365  const Eigen::array<int, 3> permute_ij_ji({1, 0});
366  const Eigen::array<int, 3> permute_ijk_jik({1, 0, 2});
367 
368  // get kinematic tensors from SystemData class
369  Eigen::Tensor<double, 1> xi_dot = TensorCast(m_system->velocitiesBodies(), m_7M); // (7M x 1)
370  Eigen::Tensor<double, 1> xi_ddot = TensorCast(m_system->accelerationsBodies(), m_7M); // (7M x 1)
371 
372  Eigen::Tensor<double, 1> V = TensorCast(m_system->velocitiesParticlesArticulation(), m_7N); // (7N x 1)
373  Eigen::Tensor<double, 1> V_dot = TensorCast(m_system->accelerationsParticlesArticulation(), m_7N); // (7N x 1)
374 
375  // calculate 2nd order kinematic tensors
376  Eigen::Tensor<double, 2> V_V = Eigen::Tensor<double, 2>(m_7N, m_7N); // (7N x 7N)
377  V_V.device(device) = V.contract(V, outer_product);
378 
379  Eigen::Tensor<double, 2> xi_dot_xi_dot = Eigen::Tensor<double, 2>(m_7M, m_7M); // (7M x 7M)
380  xi_dot_xi_dot.device(device) = xi_dot.contract(xi_dot, outer_product);
381 
382  Eigen::Tensor<double, 2> xi_dot_V = Eigen::Tensor<double, 2>(m_7M, m_7N); // (7M x 7N)
383  xi_dot_V.device(device) = xi_dot.contract(V, outer_product);
384 
385  Eigen::Tensor<double, 2> V_xi_dot = Eigen::Tensor<double, 2>(m_7N, m_7M); // (7N x 7M)
386  V_xi_dot.device(device) = xi_dot_V.shuffle(permute_ij_ji);
387 
388  // hydrodynamic forces arising from locater point motion (3 terms; contains locater inertia term)
389  Eigen::Tensor<double, 1> inertia_loc = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
390  inertia_loc.device(device) = -m_M3.contract(xi_ddot, contract_ij_j);
391 
392  Eigen::Tensor<double, 1> F_loc = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
393  F_loc.device(device) = inertia_loc;
394  F_loc.device(device) += 0.50 * m_N3.contract(xi_dot_xi_dot, contract_jki_jk);
395  F_loc.device(device) -= m_N3.contract(xi_dot_xi_dot, contract_ijk_jk);
396 
397  // hydrodynamic forces arising from coupling of locater and internal D.o.F. motion (2 terms)
398  Eigen::Tensor<double, 1> F_loc_int = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
399  F_loc_int.device(device) = m_N2.contract(xi_dot_V, contract_jki_jk);
400  F_loc_int.device(device) -= m_N2.contract(V_xi_dot, contract_ijk_jk);
401 
402  // hydrodynamic forces arising from internal D.o.F. motion (2 terms; contains internal inertia term)
403  // FIXME: This is generating a non-zero 7th element. WHY???
404  Eigen::Tensor<double, 1> F_int = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
405  F_int.device(device) = 0.50 * m_N1.contract(V_V, contract_jki_jk); // REVIEW: This is the source of the problem
406  F_int.device(device) -= m_M2.contract(V_dot, contract_ij_j);
407 
408  // compute complete potential flow hydrodynamic force
409  Eigen::Tensor<double, 1> F_hydro = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
410  F_hydro.device(device) = F_loc + F_loc_int;
411  F_hydro.device(device) += F_int;
412  m_F_hydro.noalias() = MatrixCast(F_hydro, m_7M, 1, device);
413 
414  // Compute non-inertial part of force (include internal D.o.F. inertia)
415  Eigen::Tensor<double, 1> F_hydro_no_inertia = Eigen::Tensor<double, 1>(m_7M); // (7M x 1)
416  F_hydro_no_inertia.device(device) = F_hydro - inertia_loc;
417  m_F_hydroNoInertia.noalias() = MatrixCast(F_hydro_no_inertia, m_7M, 1, device);
418 
419  // debugging print statements
420  Eigen::IOFormat CleanFmt(4, 0, ", ", "\n", "[", "]");
421  Eigen::VectorXd f1 = MatrixCast(0.50 * m_N1.contract(V_V, contract_jki_jk), m_7M, 1, device);
422  Eigen::VectorXd f2 = MatrixCast(-m_M2.contract(V_dot, contract_ij_j), m_7M, 1, device);
423 
424  std::cout << "F_int:\n" << MatrixCast(F_int, m_7M, 1, device).format(CleanFmt) << "\n\n" << std::endl;
425  std::cout << "F_int 1:\n" << f1.format(CleanFmt) << "\n\n" << std::endl;
426  std::cout << "F_int 2:\n" << f2.format(CleanFmt) << "\n\n" << std::endl;
427 }
